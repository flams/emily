  7 define(["Observable", "Tools"],
  8 /**
  9  * @class
 10  * Store creates an observable structure based on a key/values object
 11  * or on an array
 12  */
 13  function Store(Observable, Tools) {
 14 
 15 	/**
 16 	 * Defines the Store
 17 	 * @param {Array/Object} the data to initialize the store with
 18 	 * @returns
 19 	 */
 20 	return function StoreConstructor($data) {
 21 
 22 		/**
 23 		 * Where the data is stored
 24 		 * @private
 25 		 */
 26 		var _data = Tools.clone($data) || {},
 27 
 28 		/**
 29 		 * The observable for publishing changes on the store iself
 30 		 * @private
 31 		 */
 32 		_storeObservable = new Observable(),
 33 
 34 		/**
 35 		 * The observable for publishing changes on a value
 36 		 * @private
 37 		 */
 38 		_valueObservable = new Observable(),
 39 
 40 		/**
 41 		 * Gets the difference between two objects and notifies them
 42 		 * @private
 43 		 * @param {Object} previousData
 44 		 */
 45 		_notifyDiffs = function _notifyDiffs(previousData) {
 46 			var diffs = Tools.objectsDiffs(previousData, _data);
 47 			["updated",
 48 			 "deleted",
 49 			 "added"].forEach(function (value) {
 50 				 diffs[value].forEach(function (dataIndex) {
 51 						_storeObservable.notify(value, dataIndex, _data[dataIndex]);
 52 						_valueObservable.notify(dataIndex, _data[dataIndex], value);
 53 				 });
 54 			});
 55 		};
 56 
 57 		/**
 58 		 * Get the number of items in the store
 59 		 * @returns {Number} the number of items in the store
 60 		 */
 61 		this.getNbItems = function() {
 62 			return _data instanceof Array ? _data.length : Tools.count(_data);
 63 		};
 64 
 65 		/**
 66 		 * Count is an alias for getNbItems
 67 		 * @returns {Number} the number of items in the store
 68 		 */
 69 		this.count = this.getNbItems;
 70 
 71 		/**
 72 		 * Get a value from its index
 73 		 * @param {String} name the name of the index
 74 		 * @returns the value
 75 		 */
 76 		this.get = function get(name) {
 77 			return _data[name];
 78 		};
 79 
 80 		/**
 81 		 * Checks if the store has a given value
 82 		 * @param {String} name the name of the index
 83 		 * @returns {Boolean} true if the value exists
 84 		 */
 85 		this.has = function has(name) {
 86 			return _data.hasOwnProperty(name);
 87 		};
 88 
 89 		/**
 90 		 * Set a new value and overrides an existing one
 91 		 * @param {String} name the name of the index
 92 		 * @param value the value to assign
 93 		 * @returns true if value is set
 94 		 */
 95 		this.set = function set(name, value) {
 96 			var hasPrevious,
 97 				previousValue,
 98 				action;
 99 
100 			if (typeof name != "undefined") {
101 				hasPrevious = this.has(name);
102 				previousValue = this.get(name);
103 				_data[name] = value;
104 				action = hasPrevious ? "updated" : "added";
105 				_storeObservable.notify(action, name, _data[name], previousValue);
106 				_valueObservable.notify(name, _data[name], action, previousValue);
107 				return true;
108 			} else {
109 				return false;
110 			}
111 		};
112 
113 		/**
114 		 * Update the property of an item.
115 		 * @param {String} name the name of the index
116 		 * @param {String} property the property to modify.
117 		 * @param value the value to assign
118 		 * @returns false if the Store has no name index
119 		 */
120 		this.update = function update(name, property, value) {
121 			var item;
122 			if (this.has(name)) {
123 				item = this.get(name);
124 				Tools.setNestedProperty(item, property, value);
125 				_storeObservable.notify("updated", property, value);
126 				_valueObservable.notify(name, item, "updated");
127 				return true;
128 			} else {
129 				return false;
130 			}
131 		};
132 
133 		/**
134 		 * Delete value from its index
135 		 * @param {String} name the name of the index from which to delete the value
136 		 * @returns true if successfully deleted.
137 		 */
138 		this.del = function del(name) {
139 			if (this.has(name)) {
140 				if (!this.alter("splice", name, 1)) {
141 					delete _data[name];
142 					_storeObservable.notify("deleted", name);
143 					_valueObservable.notify(name, _data[name], "deleted");
144 				}
145 				return true;
146 			} else {
147 				return false;
148 			}
149 		};
150 
151 		/**
152 		 * Delete multiple indexes. Prefer this one over multiple del calls.
153 		 * @param {Array}
154 		 * @returns false if param is not an array.
155 		 */
156 		this.delAll = function delAll(indexes) {
157 			if (indexes instanceof Array) {
158 				// Indexes must be removed from the greatest to the lowest
159 				// To avoid trying to remove indexes that don't exist.
160 				// i.e: given [0, 1, 2], remove 1, then 2, 2 doesn't exist anymore
161 				indexes.sort(Tools.compareNumbers)
162 					.reverse()
163 					.forEach(this.del, this);
164 				return true;
165 			} else {
166 				return false;
167 			}
168 		};
169 
170 		/**
171 		 * Alter the data be calling one of it's method
172 		 * When the modifications are done, it notifies on changes.
173 		 * @param {String} func the name of the method
174 		 * @returns the result of the method call
175 		 */
176 		this.alter = function alter(func) {
177 			var apply,
178 				previousData;
179 
180 			if (_data[func]) {
181 				previousData = Tools.clone(_data);
182 				apply = _data[func].apply(_data, Array.prototype.slice.call(arguments, 1));
183 				_notifyDiffs(previousData);
184 				return apply;
185 			} else {
186 				return false;
187 			}
188 		};
189 
190 		/**
191 		 * proxy is an alias for alter
192 		 */
193 		 this.proxy = this.alter;
194 
195 		/**
196 		 * Watch the store's modifications
197 		 * @param {String} added/updated/deleted
198 		 * @param {Function} func the function to execute
199 		 * @param {Object} scope the scope in which to execute the function
200 		 * @returns {Handle} the subscribe's handler to use to stop watching
201 		 */
202 		this.watch = function watch(name, func, scope) {
203 			return _storeObservable.watch(name, func, scope);
204 		};
205 
206 		/**
207 		 * Unwatch the store modifications
208 		 * @param {Handle} handle the handler returned by the watch function
209 		 * @returns
210 		 */
211 		this.unwatch = function unwatch(handle) {
212 			return _storeObservable.unwatch(handle);
213 		};
214 
215 		/**
216 		 * Get the observable used for watching store's modifications
217 		 * Should be used only for debugging
218 		 * @returns {Observable} the Observable
219 		 */
220 		this.getStoreObservable = function getStoreObservable() {
221 			return _storeObservable;
222 		};
223 
224 		/**
225 		 * Watch a value's modifications
226 		 * @param {String} name the name of the value to watch for
227 		 * @param {Function} func the function to execute
228 		 * @param {Object} scope the scope in which to execute the function
229 		 * @returns handler to pass to unwatchValue
230 		 */
231 		this.watchValue = function watchValue(name, func, scope) {
232 			return _valueObservable.watch(name, func, scope);
233 		};
234 
235 		/**
236 		 * Unwatch the value's modifications
237 		 * @param {Handler} handler the handler returned by the watchValue function
238 		 * @private
239 		 * @returns true if unwatched
240 		 */
241 		this.unwatchValue = function unwatchValue(handler) {
242 			return _valueObservable.unwatch(handler);
243 		};
244 
245 		/**
246 		 * Get the observable used for watching value's modifications
247 		 * Should be used only for debugging
248 		 * @private
249 		 * @returns {Observable} the Observable
250 		 */
251 		this.getValueObservable = function getValueObservable() {
252 			return _valueObservable;
253 		};
254 
255 		/**
256 		 * Loop through the data
257 		 * @param {Function} func the function to execute on each data
258 		 * @param {Object} scope the scope in wich to run the callback
259 		 */
260 		this.loop = function loop(func, scope) {
261 			Tools.loop(_data, func, scope);
262 		};
263 
264 		/**
265 		 * Reset all data and get notifications on changes
266 		 * @param {Arra/Object} data the new data
267 		 * @returns {Boolean}
268 		 */
269 		this.reset = function reset(data) {
270 			if (data instanceof Object) {
271 				var previousData = Tools.clone(_data);
272 				_data = Tools.clone(data) || {};
273 				_notifyDiffs(previousData);
274 				return true;
275 			} else {
276 				return false;
277 			}
278 
279 		};
280 
281 		/**
282 		 * Returns a JSON version of the data
283 		 * Use dump if you want all the data as a plain js object
284 		 * @returns {String} the JSON
285 		 */
286 		this.toJSON = function toJSON() {
287 			return JSON.stringify(_data);
288 		};
289 
290 		/**
291 		 * Returns the store's data
292 		 * @returns {Object} the data
293 		 */
294 		this.dump = function dump() {
295 			return _data;
296 		};
297 	};
298 });

 16     /**
 17      * Get the closest number in an array
 18      * @param {Number} item the base number
 19      * @param {Array} array the array to search into
 20      * @param {Function} getDiff returns the difference between the base number and
 21      *   and the currently read item in the array. The item which returned the smallest difference wins.
 22      * @private
 23      */
 24     function _getClosest(item, array, getDiff) {
 25         var closest,
 26             diff;
 27 
 28         if (!array) {
 29             return;
 30         }
 31 
 32         array.forEach(function (comparedItem, comparedItemIndex) {
 33             var thisDiff = getDiff(comparedItem, item);
 34 
 35             if (thisDiff >= 0 && (typeof diff == "undefined" || thisDiff < diff)) {
 36                 diff = thisDiff;
 37                 closest = comparedItemIndex;
 38             }
 39         });
 40 
 41         return closest;
 42     }

127         /**
128          * Small adapter for looping over objects and arrays
129          * Warning: it's not meant to be used with nodeList
130          * To use with nodeList, convert to array first
131          * @param {Array/Object} iterated the array or object to loop through
132          * @param {Function} callback the function to execute for each iteration
133          * @param {Object} scope the scope in which to execute the callback
134          * @returns {Boolean} true if executed
135          */
136         loop: function loop(iterated, callback, scope) {
137             var i,
138                 length;
139 
140             if (iterated instanceof Object && callback instanceof Function) {
141                 if (iterated instanceof Array) {
142                     for (i=0; i<iterated.length; i++) {
143                         callback.call(scope, iterated[i], i, iterated);
144                     }
145                 } else {
146                     for (i in iterated) {
147                         if (iterated.hasOwnProperty(i)) {
148                             callback.call(scope, iterated[i], i, iterated);
149                         }
150                     }
151                 }
152                 return true;
153             } else {
154                 return false;
155             }

158         /**
159          * Make a diff between two objects
160          * @param {Array/Object} before is the object as it was before
161          * @param {Array/Object} after is what it is now
162          * @example:
163          *  With objects:
164          *
165          *  before = {a:1, b:2, c:3, d:4, f:6}
166          *  after = {a:1, b:20, d: 4, e: 5}
167          *  will return :
168          *  {
169          *      unchanged: ["a", "d"],
170          *      updated: ["b"],
171          *      deleted: ["f"],
172          *      added: ["e"]
173          *  }
174          *
175          * It also works with Arrays:
176          *
177          *  before = [10, 20, 30]
178          *  after = [15, 20]
179          *  will return :
180          *  {
181          *      unchanged: [1],
182          *      updated: [0],
183          *      deleted: [2],
184          *      added: []
185          *  }
186          *
187          * @returns object
188          */
189         objectsDiffs : function objectsDiffs(before, after) {
190             if (before instanceof Object && after instanceof Object) {
191                 var unchanged = [],
192                     updated = [],
193                     deleted = [],
194                     added = [];
195 
196                  // Look through the after object
197                  this.loop(after, function (value, idx) {
198 
199                      // To get the added
200                      if (typeof before[idx] == "undefined") {
201                          added.push(idx);
202 
203                      // The updated
204                      } else if (value !== before[idx]) {
205                          updated.push(idx);
206 
207                      // And the unchanged
208                      } else if (value === before[idx]) {
209                          unchanged.push(idx);
210                      }
211 
212                  });
213 
214                  // Loop through the before object
215                  this.loop(before, function (value, idx) {
216 
217                     // To get the deleted
218                     if (typeof after[idx] == "undefined") {
219                         deleted.push(idx);
220                     }
221                  });
222 
223                 return {
224                     updated: updated,
225                     unchanged: unchanged,
226                     added: added,
227                     deleted: deleted
228                 };
229 
230             } else {
231                 return false;
232             }
233         },

277         /**
278          * Get the property of an object nested in one or more objects
279          * given an object such as a.b.c.d = 5, getNestedProperty(a, "b.c.d") will return 5.
280          * @param {Object} object the object to get the property from
281          * @param {String} property the path to the property as a string
282          * @returns the object or the the property value if found
283          */
284         getNestedProperty: function getNestedProperty(object, property) {
285             if (object && object instanceof Object) {
286                 if (typeof property == "string" && property !== "") {
287                     var split = property.split(".");
288                     return split.reduce(function (obj, prop) {
289                         return obj && obj[prop];
290                     }, object);
291                 } else if (typeof property == "number") {
292                     return object[property];
293                 } else {
294                     return object;
295                 }
296             } else {
297                 return object;
298             }
299         },

301         /**
302          * Set the property of an object nested in one or more objects
303          * If the property doesn't exist, it gets created.
304          * @param {Object} object
305          * @param {String} property
306          * @param value the value to set
307          * @returns object if no assignment was made or the value if the assignment was made
308          */
309         setNestedProperty: function setNestedProperty(object, property, value) {
310             if (object && object instanceof Object) {
311                 if (typeof property == "string" && property !== "") {
312                     var split = property.split(".");
313                     return split.reduce(function (obj, prop, idx) {
314                         obj[prop] = obj[prop] || {};
315                         if (split.length == (idx + 1)) {
316                             obj[prop] = value;
317                         }
318                         return obj[prop];
319                     }, object);
320                 } else if (typeof property == "number") {
321                     object[property] = value;
322                     return object[property];
323                 } else {
324                     return object;
325                 }
326             } else {
327                 return object;
328             }

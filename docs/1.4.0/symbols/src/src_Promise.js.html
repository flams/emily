  6 define(["Observable", "StateMachine"],
  7 
  8 /**
  9  * @class
 10  * Create a promise/A+
 11  */
 12 function Promise(Observable, StateMachine) {
 13 
 14     return function PromiseConstructor() {
 15 
 16         /**
 17          * The fulfilled value
 18          * @private
 19          */
 20         var _value = null,
 21 
 22         /**
 23          * The rejection reason
 24          * @private
 25          */
 26         _reason = null,
 27 
 28         /**
 29          * The funky observable
 30          * @private
 31          */
 32         _observable = new Observable(),
 33 
 34         /**
 35          * The state machine States & transitions
 36          * @private
 37          */
 38         _states = {
 39 
 40             // The promise is pending
 41             "Pending": [
 42 
 43                 // It can only be fulfilled when pending
 44                 ["fulfill", function onFulfill(value) {
 45                     _value = value;
 46                     _observable.notify("fulfill", value);
 47                 // Then it transits to the fulfilled state
 48                 }, "Fulfilled"],
 49 
 50                 // it can only be rejected when pending
 51                 ["reject", function onReject(reason) {
 52                     _reason = reason;
 53                     _observable.notify("reject", reason);
 54                 // Then it transits to the rejected state
 55                 }, "Rejected"],
 56 
 57                 // When pending, add the resolver to an observable
 58                 ["toFulfill", function toFulfill(resolver) {
 59                     _observable.watch("fulfill", resolver);
 60                 }],
 61 
 62                 // When pending, add the resolver to an observable
 63                 ["toReject", function toReject(resolver) {
 64                     _observable.watch("reject", resolver);
 65                 }]],
 66 
 67             // When fulfilled,
 68             "Fulfilled": [
 69                 // We directly call the resolver with the value
 70                 ["toFulfill", function toFulfill(resolver) {
 71                     setTimeout(function () {
 72                         resolver(_value);
 73                     }, 0);
 74                 }]],
 75 
 76             // When rejected
 77             "Rejected": [
 78                 // We directly call the resolver with the reason
 79                 ["toReject", function toReject(resolver) {
 80                     setTimeout(function () {
 81                         resolver(_reason);
 82                     }, 0);
 83                 }]]
 84         },
 85 
 86         /**
 87          * The stateMachine
 88          * @private
 89          */
 90         _stateMachine = new StateMachine("Pending", _states);
 91 
 92         /**
 93          * Fulfilled the promise.
 94          * A promise can be fulfilld only once.
 95          * @param the fulfillment value
 96          * @returns the promise
 97          */
 98         this.fulfill = function fulfill(value) {
 99             _stateMachine.event("fulfill", value);
100             return this;
101         };
102 
103         /**
104          * Reject the promise.
105          * A promise can be rejected only once.
106          * @param the rejection value
107          * @returns true if the rejection function was called
108          */
109         this.reject = function reject(reason) {
110             _stateMachine.event("reject", reason);
111             return this;
112         };
113 
114         /**
115          * The callbacks to call after fulfillment or rejection
116          * @param {Function} fulfillmentCallback the first parameter is a success function, it can be followed by a scope
117          * @param {Function} the second, or third parameter is the rejection callback, it can also be followed by a scope
118          * @examples:
119          *
120          * then(fulfillment)
121          * then(fulfillment, scope, rejection, scope)
122          * then(fulfillment, rejection)
123          * then(fulfillment, rejection, scope)
124          * then(null, rejection, scope)
125          * @returns {Promise} the new promise
126          */
127         this.then = function then() {
128             var promise = new PromiseConstructor();
129 
130             // If a fulfillment callback is given
131             if (arguments[0] instanceof Function) {
132                 // If the second argument is also a function, then no scope is given
133                 if (arguments[1] instanceof Function) {
134                     _stateMachine.event("toFulfill", this.makeResolver(promise, arguments[0]));
135                 } else {
136                     // If the second argument is not a function, it's the scope
137                     _stateMachine.event("toFulfill", this.makeResolver(promise, arguments[0], arguments[1]));
138                 }
139             } else {
140                 // If no fulfillment callback given, give a default one
141                 _stateMachine.event("toFulfill", this.makeResolver(promise, function () {
142                     promise.fulfill(_value);
143                 }));
144             }
145 
146             // if the second arguments is a callback, it's the rejection one, and the next argument is the scope
147             if (arguments[1] instanceof Function) {
148                 _stateMachine.event("toReject", this.makeResolver(promise, arguments[1], arguments[2]));
149             }
150 
151             // if the third arguments is a callback, it's the rejection one, and the next arguments is the sopce
152             if (arguments[2] instanceof Function) {
153                 _stateMachine.event("toReject", this.makeResolver(promise, arguments[2], arguments[3]));
154             }
155 
156             // If no rejection callback is given, give a default one
157             if (!(arguments[1] instanceof Function) &&
158                 !(arguments[2] instanceof Function)) {
159                 _stateMachine.event("toReject", this.makeResolver(promise, function () {
160                     promise.reject(_reason);
161                 }));
162             }
163 
164             return promise;
165         };
166 
167         /**
168          * Synchronize this promise with a thenable
169          * @returns {Boolean} false if the given sync is not a thenable
170          */
171         this.sync = function sync(syncWith) {
172             if (syncWith instanceof Object && syncWith.then) {
173 
174                 var onFulfilled = function onFulfilled(value) {
175                     this.fulfill(value);
176                 },
177                 onRejected = function onRejected(reason) {
178                     this.reject(reason);
179                 };
180 
181                 syncWith.then(onFulfilled.bind(this),
182                         onRejected.bind(this));
183 
184                 return true;
185             } else {
186                 return false;
187             }
188         };
189 
190         /**
191          * Make a resolver
192          * for debugging only
193          * @private
194          * @returns {Function} a closure
195          */
196         this.makeResolver = function makeResolver(promise, func, scope) {
197             return function resolver(value) {
198                 var returnedPromise;
199 
200                 try {
201                     returnedPromise = func.call(scope, value);
202                     if (!promise.sync(returnedPromise)) {
203                         promise.fulfill(returnedPromise);
204                     }
205                 } catch (err) {
206                     promise.reject(err);
207                 }
208 
209             };
210         };
211 
212         /**
213          * Returns the reason
214          * for debugging only
215          * @private
216          */
217         this.getReason = function getReason() {
218             return _reason;
219         };
220 
221         /**
222          * Returns the reason
223          * for debugging only
224          * @private
225          */
226         this.getValue = function getValue() {
227             return _value;
228         };
229 
230         /**
231          * Get the promise's observable
232          * for debugging only
233          * @private
234          * @returns {Observable}
235          */
236         this.getObservable = function getObservable() {
237             return _observable;
238         };
239 
240         /**
241          * Get the promise's stateMachine
242          * for debugging only
243          * @private
244          * @returns {StateMachine}
245          */
246         this.getStateMachine = function getStateMachine() {
247             return _stateMachine;
248         };
249 
250         /**
251          * Get the statesMachine's states
252          * for debugging only
253          * @private
254          * @returns {Object}
255          */
256         this.getStates = function getStates() {
257             return _states;
258         };
259 
260     };
261 
262 
263 
264 
265 });

 11 /**
 12  * @class
 13  * CouchDBStore synchronises a Store with a CouchDB view or document
 14  * It subscribes to _changes to keep its data up to date.
 15  */

 18 	/**
 19 	 * Defines the CouchDBStore

 24 		/**
 25 		 * The name of the channel on which to run the requests
 26 		 * @private
 27 		 */
 28 		var _channel = "CouchDB",
 29 		
 30 		/**
 31 		 * The transport used to run the requests
 32 		 * @private
 33 		 */
 34 		_transport = null,
 35 		
 36 		/**
 37 		 * That will store the synchronization info
 38 		 * @private
 39 		 */
 40 		_syncInfo = {},
 41 		
 42 		/**
 43 		 * The promise that is returned by sync
 44 		 * It's resolved when entering listening state
 45 		 * It's rejected when no such document to sync to
 46 		 * @private
 47 		 */
 48 		_syncPromise = new Promise(),
 49 		
 50 		/**
 51 		 * All the actions performed by the couchDBStore
 52 		 * They'll feed the stateMachine
 53 		 * @private
 54 		 */

 57 			/**
 58 			 * Get a CouchDB view
 59 			 * @private
 60 			 */
 61 			getView: function () {
 62 
 63 				_syncInfo.query = _syncInfo.query || {};
 64 				_syncInfo.query.update_seq=true;
 65 				
 66 				_transport.request(_channel, {
 67 					method: "GET",
 68 					path: "/" + _syncInfo.database + "/_design/" + _syncInfo.design + "/" + _syncInfo.view,
 69 					query: _syncInfo.query
 70 				}, function (results) {
 71 					var json = JSON.parse(results);
 72 					if (!json.rows) {
 73 						throw new Error("CouchDBStore [" + _syncInfo.database + ", " + _syncInfo.design + ", " + _syncInfo.view + "].sync() failed: " + results);	
 74 					} else {
 75 						this.reset(json.rows);
 76 						
 77 						if (typeof json.total_rows == "undefined") {
 78 							this.setReducedViewInfo(true);
 79 						}
 80 
 81 						_stateMachine.event("subscribeToViewChanges", json.update_seq);
 82 					}
 83 				}, this);
 84 			},
 85 			
 86 			/**
 87 			 * Get a CouchDB document
 88 			 * @private
 89 			 */
 90 			getDocument: function () { 
 91 				
 92 				_transport.request(_channel, {
 93 					method: "GET",
 94 					path: "/" + _syncInfo.database + "/" + _syncInfo.document,
 95 					query: _syncInfo.query
 96 				}, function (results) {
 97 					var json = JSON.parse(results);
 98 					if (json._id) {
 99 						this.reset(json);
100 						_stateMachine.event("subscribeToDocumentChanges");	
101 					} else {
102 						_syncPromise.reject(this);
103 					}
104 				}, this);
105 			},

107 			/**
108 			 * Get a bulk of documents
109 			 * @private
110 			 */
111 			getBulkDocuments: function (lol) {
112 				
113 				var reqData = {
114 							path: "/" + _syncInfo.database + "/_all_docs",
115 							query: _syncInfo.query
116 						},
117 						errorString;
118 
119 				// If an array of keys is defined, we POST it to _all_docs to get arbitrary docs.
120 				if (_syncInfo["keys"] instanceof Array) {
121 					reqData.method = "POST";
122 					reqData.data = JSON.stringify({keys:_syncInfo.keys});
123 					reqData.headers = {
124 						"Content-Type": "application/json"
125 					};
126 					errorString = reqData.data;
127 					
128 				// Else, we just GET the documents using startkey/endkey
129 				} else {
130 					reqData.method = "GET";
131 					errorString = JSON.stringify(_syncInfo.query);
132 				}
133 				
134 				_syncInfo.query.include_docs = true;
135 				_syncInfo.query.update_seq = true;
136 				
137 				_transport.request(_channel, 
138 					reqData,
139 					function (results) {
140 					
141 					var json = JSON.parse(results);
142 					
143 					if (!json.rows) {
144 						throw new Error("CouchDBStore.sync(\"" + _syncInfo.database + "\", " + errorString + ") failed: " + results);	
145 					} else {
146 						this.reset(json.rows);
147 						_stateMachine.event("subscribeToBulkChanges", json.update_seq);
148 					}
149 				}, this);
150 				
151 			},

153 			/**
154 			 * Put a new document in CouchDB
155 			 * @private
156 			 */
157 			createDocument: function (promise) {
158             	_transport.request(_channel, {
159             		method: "PUT",
160             		path: "/" + _syncInfo.database + "/" + _syncInfo.document,
161             		headers: {
162             			"Content-Type": "application/json"
163             		},
164             		data: this.toJSON()
165             	}, function (result) {
166             		var json = JSON.parse(result);
167             		if (json.ok) {
168             			promise.resolve(json);
169                 		_stateMachine.event("subscribeToDocumentChanges");	
170             		} else {
171             			promise.reject(json);
172             		}
173             	});
174             },

176             /**
177              * Subscribe to changes when synchronized with a view
178              * @param {Number} the update_seq given by getView, it'll be passed to since in the GET request
179              * @private
180              */
181             subscribeToViewChanges: function (update_seq) {
182             	
183             	Tools.mixin({
184 					feed: "continuous",
185 					heartbeat: 20000,
186 					since: update_seq
187 				}, _syncInfo.query);
188             	
189             	this.stopListening = _transport.listen(_channel, {
190 						path: "/" + _syncInfo.database + "/_changes",
191 						query: _syncInfo.query
192 					},
193 					function (changes) {
194 						// Should I test for this very special case (heartbeat?)
195 						// Or do I have to try catch for any invalid json?
196 						if (changes == "\n") {
197 							return false;
198 						}
199 						
200 						var json = JSON.parse(changes),
201 							action;
202 						
203 						// reducedView is known on the first get view
204 						if (_syncInfo.reducedView) { 
205 							action = "updateReduced";
206 						} else {
207 							if (json.deleted) {
208 								action = "delete";
209 							} else if (json.changes[0].rev.search("1-") == 0) {
210 								action = "add";
211 							} else {
212 								action = "change";
213 							}
214 						}
215 							
216 						_stateMachine.event(action, json.id);
217 					}, this);
218 			},
219 			
220 			/**
221 			 * Subscribe to changes when synchronized with a document
222 			 * @private
223 			 */
224 			subscribeToDocumentChanges: function () {
225 
226 				this.stopListening = _transport.listen(_channel, {
227 					path: "/" + _syncInfo.database + "/_changes",
228 					query: {
229 						 feed: "continuous",
230 						 heartbeat: 20000
231 						}
232 					},
233 				function (changes) {
234 					var json;
235 					// Should I test for this very special case (heartbeat?)
236 					// Or do I have to try catch for any invalid json?
237 					if (changes == "\n") {
238 						return false;
239 					}
240 					
241 					json = JSON.parse(changes);
242 					
243 					// The document is the modified document is the current one
244 					if (json.id == _syncInfo.document && 
245 						// And if it has a new revision
246 						json.changes.pop().rev != this.get("_rev")) {
247 						
248 						if (json.deleted) {
249 							_stateMachine.event("deleteDoc");
250 						} else {
251 							_stateMachine.event("updateDoc");	
252 						}
253 					 }
254 				}, this);
255 			},

257 			/**
258 			 * Subscribe to changes when synchronized with a bulk of documents
259 			 * @private
260 			 */
261 			subscribeToBulkChanges: function (update_seq) {
262 				Tools.mixin({
263 					feed: "continuous",
264 					heartbeat: 20000,
265 					since: update_seq,
266 					include_docs: true
267 				}, _syncInfo.query);
268             	
269             	this.stopListening = _transport.listen(_channel, {
270 						path: "/" + _syncInfo.database + "/_changes",
271 						query: _syncInfo.query
272 					},
273 					function (changes) {
274 						var json;
275 						// Should I test for this very special case (heartbeat?)
276 						// Or do I have to try catch for any invalid json?
277 						if (changes == "\n") {
278 							return false;
279 						}
280 						
281 						var json = JSON.parse(changes),
282 							action;
283 						
284 						if (json.changes[0].rev.search("1-") == 0) {
285 							action = "bulkAdd";
286 						} else if (json.deleted) {
287 							action = "delete";
288 						} else {
289 							action = "bulkChange";
290 						}
291 						
292 						_stateMachine.event(action, json.id, json.doc);
293 						
294 						
295 					}, this);
296 			},

298 			/**
299 			 * Update in the Store a document that was updated in CouchDB
300 			 * Get the whole view :(, then get the modified document and update it.
301 			 * I have no choice but to request the whole view and look for the document
302 			 * so I can also retrieve its position in the store (idx) and update the item.
303 			 * Maybe I've missed something
304 			 * @private
305 			 */
306 			updateDocInStore: function (id) {
307 				_transport.request(_channel,{
308 					method: "GET",
309 					path: "/" + _syncInfo.database + "/_design/" + _syncInfo.design + "/" + _syncInfo.view,
310 					query: _syncInfo.query
311 				}, function (view) {
312 					var json = JSON.parse(view);
313 					
314 					if (json.rows.length == this.getNbItems()) {
315 						json.rows.some(function (value, idx) {
316 							if (value.id == id) {
317 								this.set(idx, value);
318 							}
319 						}, this);
320 					} else {
321 						this.actions.evenDocsInStore.call(this, json.rows, id);
322 					}
323 
324 				}, this);
325 				
326 			},
327 			
328 			/**
329 			 * When a doc is removed from the view even though it still exists
330 			 * or when it's added to a view, though it wasn't just created
331 			 * This function must be called to even the store
332 			 * @private
333 			 */
334 			evenDocsInStore: function (view, id) {
335 				var nbItems = this.getNbItems();
336 				
337 				// If a document was removed from the view
338 				if (view.length < nbItems) {
339 					
340 					// Look for it in the store to remove it
341 					this.loop(function (value, idx) {
342 						if (value.id == id) {
343 							this.del(idx);
344 						}
345 					}, this);
346 					
347 				// If a document was added to the view
348 				} else if (view.length > nbItems) {
349 					
350 					// Look for it in the view and add it to the store at the same place
351 					view.some(function (value, idx) {
352 						if (value.id == id) {
353 							return this.alter("splice", idx, 0, value);
354 						}
355 					}, this);
356 
357 				}
358 
359 			},
360 			
361 			/**
362 			 * Add in the Store a document that was added in CouchDB
363 			 * @private
364 			 */
365 			addBulkDocInStore: function (id) {
366 				if (_syncInfo["query"].startkey || _syncInfo["query"].endkey) {
367 					_syncInfo.query.include_docs = true;
368 					_syncInfo.query.update_seq = true;
369 				
370 					_transport.request(_channel, {
371 						method: "GET",
372 						path: "/" + _syncInfo.database + "/_all_docs",
373 						query: _syncInfo.query
374 					},
375 					function (results) {
376 						
377 						var json = JSON.parse(results);
378 						
379 						json.rows.forEach(function (value, idx) {
380 							if (value.id == id) {
381 								this.alter("splice", idx, 0, value.doc);
382 							}
383 						}, this);
384 	
385 					}, this);
386 				} else {
387 					return false;
388 				}
389 			},

554 		/**
555 		 * The state machine
556 		 * @private
557 		 * it concentrates almost the whole logic.
558 		 */
559 		_stateMachine = new StateMachine("Unsynched", {
560 			"Unsynched": [
561 			    ["getView", actions.getView, this, "Synched"],
562 				["getDocument", actions.getDocument, this, "Synched"],
563 				["getBulkDocuments", actions.getBulkDocuments, this, "Synched"]
564 			 ],
565 						
566 			"Synched": [
567 			    ["updateDatabase", actions.createDocument, this],
568 			    ["subscribeToViewChanges", actions.subscribeToViewChanges, this, "Listening"],
569 				["subscribeToDocumentChanges", actions.subscribeToDocumentChanges, this, "Listening"],
570 				["subscribeToBulkChanges", actions.subscribeToBulkChanges, this, "Listening"],
571 				["unsync", function noop(){}, "Unsynched"]
572 			 ],
573 				
574 			"Listening": [
575 			    ["entry", actions.resolve, this],
576 			    ["change", actions.updateDocInStore, this],
577 			    ["bulkAdd", actions.addBulkDocInStore, this],
578 			    ["bulkChange", actions.updateBulkDocInStore, this],
579 				["delete", actions.removeDocInStore, this],
580 				["add", actions.addDocInStore, this],
581 				["updateReduced", actions.updateReduced, this],
582 				["updateDoc", actions.updateDoc, this],
583 			    ["deleteDoc", actions.deleteDoc, this],
584 			    ["updateDatabase", actions.updateDatabase, this],
585 			    ["updateDatabaseWithBulkDoc", actions.updateDatabaseWithBulkDoc, this],
586 			    ["removeFromDatabase", actions.removeFromDatabase, this],
587 			    ["unsync", actions.unsync, this, "Unsynched"]
588 			   ]

592 		/**
593 		 * Synchronize the store with a view
594 		 * @param {String} database the name of the database where to get...
595 		 * @param {String} ...design the design document, in which...
596 		 * @param {String} view ...the view is.
597 		 * @returns {Boolean}
598 		 */
599 		this.sync = function sync() {
600 			if (typeof arguments[0] == "string" && typeof arguments[1] == "string" && typeof arguments[2] == "string") {
601 				this.setSyncInfo(arguments[0], arguments[1], arguments[2], arguments[3]);
602 				_stateMachine.event("getView");
603 				return _syncPromise;
604 			} else if (typeof arguments[0] == "string" && typeof arguments[1] == "string" && typeof arguments[2] != "string") {
605 				this.setSyncInfo(arguments[0], arguments[1], arguments[2]);
606 				_stateMachine.event("getDocument");
607 				return _syncPromise;
608 			} else if (typeof arguments[0] == "string" && arguments[1] instanceof Object) {
609 				this.setSyncInfo(arguments[0], arguments[1]);
610 				_stateMachine.event("getBulkDocuments");
611 				return _syncPromise;
612 			}
613 			return false;
614 		};
615 		
616 		/**
617 		 * Set the synchronization information
618 		 * @private
619 		 * @returns {Boolean}
620 		 */
621 		this.setSyncInfo = function setSyncInfo() {
622 			this.clearSyncInfo();
623 			if (typeof arguments[0] == "string" && typeof arguments[1] == "string" && typeof arguments[2] == "string") {
624 				_syncInfo["database"] = arguments[0];
625 				_syncInfo["design"] = arguments[1];
626 				_syncInfo["view"] = arguments[2];
627 				_syncInfo["query"] = arguments[3];
628 				return true;
629 			} else if (typeof arguments[0] == "string" && typeof arguments[1] == "string" && typeof arguments[2] != "string") {
630 				_syncInfo["database"] = arguments[0];
631 				_syncInfo["document"] = arguments[1];
632 				_syncInfo["query"] = arguments[2];
633 				return true;
634 			} else if (typeof arguments[0] == "string" && arguments[1] instanceof Object) {
635 				_syncInfo["database"] = arguments[0];
636 				_syncInfo["query"] = arguments[1];
637 				if (_syncInfo["query"].keys instanceof Array) {
638 					_syncInfo["keys"] = _syncInfo["query"].keys;
639 					delete _syncInfo["query"].keys;
640 				}
641 				return true;
642 			}
643 			return false;
644 		};

653 		};

687 		/**
688 		 * Upload the document to the database
689 		 * Works for CouchDBStore that are synchronized with documents or bulk of documents.

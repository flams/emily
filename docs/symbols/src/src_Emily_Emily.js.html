1 /**
  2  * @class
  3  * Emily is the module loader.
  4  * You can declare modules and require them.
  5  * There's a testing mode that allows modules to run in isolation.
  6  * A module that requires an other module will get an injected
  7  * one instead. 
  8  * Emily can also serve modules that are globally defined,
  9  * so they can be stubbed or mocked. 
 10  */
 11 function Emily() {
 12 	
 13 	/**
 14 	 * The list of modules' constructors
 15 	 * @private
 16 	 */
 17 	var _constructors = {},
 18 	_modules = {},
 19 	
 20 	/**
 21 	 * The list of injected modules while in isolationMode 
 22 	 * @private
 23 	 */
 24 	_injected = {},
 25 		
 26 	/**
 27 	 * The isolationMode boolean. While in isolationMode,
 28 	 * only the injected modules are returned by require()
 29 	 * @private 
 30 	 */
 31 	_isolationMode = false;
 32 	
 33     /**
 34      * Declare a new module
 35      * @param {String} name the name of the module
 36      * @param {String} inherits - optional the name of the module to inherit from
 37      * @param {Function} func the module's constructor
 38      * @returns the module's constructor
 39      */
 40     this.declare = function declare(name) {   
 41     	var func;
 42     	
 43     	if (typeof arguments[1] == "string" && typeof arguments[2] == "function") {
 44     		func = arguments[2];
 45     		func.prototype = new _constructors[arguments[1]](this);
 46     	} else if (typeof arguments[1] == "function"){
 47     		func = arguments[1];
 48     	} else {
 49     		return false;
 50     	}
 51     	
 52     	func.prototype._declaredName = name;
 53     	
 54     	_modules[name] = new func(this);
 55     	return _constructors[name] = func;
 56     };
 57     
 58     /**
 59      * Require a module
 60      * @param {String} name the name of the module
 61      * @returns {Object} the module's constructor
 62      */
 63     this.require = function require(name) {
 64 		return _isolationMode && _injected[name] || _modules[name] || this.require("Tools").getGlobal()[name];
 65     };
 66     
 67     /**
 68      * Set isolationMode. In unit testing, modules can run in isolation.
 69      * To inject a test double, use the inject function.
 70      * @param {Boolean} the isolationMode value
 71      */
 72     this.setIsolationMode = function setIsolationMode(value) {
 73     	_isolationMode = value;
 74     };
 75     
 76     /**
 77      * Know the unit testing mode value
 78      * @returns {Boolean}
 79      */
 80     this.getIsolationMode = function getIsolationMode() {
 81     	return _isolationMode;
 82     };
 83     
 84     /**
 85      * Inject a test double
 86      * @param {String} name the name of the module to double
 87      * @param {Object} obj the test double
 88      */
 89     this.inject = function inject(name, obj) {
 90     	_injected[name] = obj;
 91     };
 92 
 93     
 94     
 95 };
 96 
 97 var Emily = new Emily();

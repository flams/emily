1 /**
  2  * @class
  3  */
  4 function Emily() {
  5 	
  6 	/**
  7 	 * The list of services
  8 	 * @private
  9 	 */
 10 	var _services = {},
 11 		_injected = {},
 12 		_isolationMode = false;
 13 	
 14     /**
 15      * Declare a new service
 16      * @param {String} name the name of the service
 17      * @param {Function} func the service's function
 18      */
 19     this.declare = function declare(name, func) {   	
 20     	_services[name] = new func(this);
 21     	return true;
 22     };
 23     
 24     /**
 25      * Require a service
 26      * @param {String} name the name of the service
 27      * @returns {Object} the service
 28      */
 29     this.require = function require(name) {
 30     	if (_isolationMode) {
 31     		return _injected[name];
 32     	} else {
 33     		return _services[name] || window[name];	
 34     	}
 35     };
 36     
 37     /**
 38      * Switch to unit testing. In unit testing, services run in isolation.
 39      * To stubb a service, use the inject function
 40      * @param value
 41      */
 42     this.setIsolationMode = function setIsolationMode(value) {
 43     	_isolationMode = value;
 44     };
 45     
 46     /**
 47      * Know the unit testing mode value
 48      * @returns {Boolean}
 49      */
 50     this.getIsolationMode = function getIsolationMode() {
 51     	return _isolationMode;
 52     };
 53     
 54     /**
 55      * Inject a stubbed service
 56      * @param {String} name the name of the service to stubb
 57      * @param {Object} obj the stubbed service
 58      */
 59     this.inject = function inject(name, obj) {
 60     	_injected[name] = obj;
 61     };
 62     
 63     
 64 };
 65 
 66 var Emily = new Emily;

1 define("Tools",
  2 /**
  3  * @class
  4  * Tools is a collection of tools 
  5  */
  6 function Tools (){
  7 
  8 	return {
  9 	    /**
 10 	     * For applications that don't run in a browser, window is not the global object.
 11 	     * This function returns the global object wherever the application runs.
 12 	     * @returns {Object} the global object
 13 	     */
 14 		getGlobal: function getGlobal() {
 15 	    	var func = function() {
 16 	    		return this;
 17 	    	};
 18 	    	return func.call(null);
 19 	    },
 20 		
 21 		/**
 22 		 * Mixes an object into another
 23 		 * @param {Object} destination object to mix values into
 24 		 * @param {Object} source object to get values from
 25 		 * @param {Boolean} optional, set to true to prevent overriding
 26 		 */
 27 	    mixin: function mixin(source, destination, dontOverride) {
 28 			this.loop(source, function (value, idx) {
 29 				if (!destination[idx] || !dontOverride) {
 30 					destination[idx] = source[idx];	
 31 				}
 32 			});
 33 			return destination;
 34 		},
 35 		
 36 		/**
 37 		 * Count the number of properties in an object
 38 		 * It doesn't look up in the prototype chain
 39 		 * @param {Object} object the object to count
 40 		 * @returns {Number}
 41 		 */
 42 		count: function count(object) {
 43 			var nbItems = 0;
 44 			this.loop(object, function () {
 45 				nbItems++;
 46 			});
 47 			
 48 			return nbItems;
 49 		},
 50 		
 51 		/**
 52 		 * Compares the properties of two objects and returns true if they're the same
 53 		 * It's doesn't do it recursively
 54 		 * @param {Object} first object
 55 		 * @param {Object} second object
 56 		 * @returns {Boolean} true if the two objets have the same properties
 57 		 */
 58 		compareObjects: function compareObjects(object1, object2) {
 59 			var getOwnProperties = function (object) {
 60 				return Object.getOwnPropertyNames(object).sort().join("");
 61 			};
 62 			return getOwnProperties(object1) == getOwnProperties(object2);
 63 		},
 64 		
 65 		/**
 66 		 * Transform array-like objects to array, such as nodeLists or arguments
 67 		 * @param {Array-like object}
 68 		 * @returns {Array}
 69 		 */
 70 		toArray: function toArray(array) {
 71 			return Array.prototype.slice.call(array);
 72 		},
 73 		
 74 		/**
 75 		 * Small adapter for looping over objects and arrays
 76 		 * @param {Array/Object} iterated the array or object to loop through
 77 		 * @param {Function} callback the function to execute for each iteration
 78 		 * @param {Object} scope the scope in which to execute the callback
 79 		 * @returns {Boolean} true if executed
 80 		 */
 81 		loop: function loop(iterated, callback, scope) {
 82 			var i;
 83 			
 84 			if (iterated instanceof Object && typeof callback == "function") {
 85 				if (iterated instanceof Array) {
 86 					iterated.forEach(callback, scope);
 87 				} else {
 88 					for (i in iterated) {
 89 						if (iterated.hasOwnProperty(i)) {
 90 							callback.call(scope, iterated[i], i, iterated);
 91 						}
 92 					}
 93 				}
 94 				return true;
 95 			} else {
 96 				return false;
 97 			}
 98 		},
 99 		
100 		objectsDiffs : function objectsDiffs(before, after) {
101 			if (before instanceof Object && after instanceof Object) {
102 				var unchanged = [],
103 					updated = [],
104 					deleted = [],
105 					added = [];
106 				
107 				 // Look for the unchanged values
108 				 this.loop(after, function (value, idx) {
109 					 if (value !== before[idx] && typeof before[idx] != "undefined") {
110 						 updated.push(idx);
111 					 } else if (value === before[idx]) {
112 						 unchanged.push(idx);
113 					 } else if (typeof before[idx] == "undefined") {
114 						 added.push(idx);
115 					 }
116 				 });
117 				 
118 				 // Look for the deleted
119 				 this.loop(before, function (value, idx) {
120 					if (typeof after[idx] == "undefined") {
121 						deleted.push(idx);
122 					} 
123 				 });
124 				 
125 				return {
126 					updated: updated,
127 					unchanged: unchanged,
128 					added: added,
129 					deleted: deleted
130 				};
131 				
132 			} else {
133 				return false;
134 			}
135 		},
136 		
137 		/**
138 		 * Transforms Arrays and Objects into valid JSON
139 		 * @param {Object/Array} object the object to JSONify 
140 		 * @returns the JSONified object or false if failed
141 		 */
142 		jsonify: function jsonify(object) {
143 			if (object instanceof Object) {
144 				return JSON.parse(JSON.stringify(object));	
145 			} else {
146 				return false;
147 			}
148 		},
149 		
150 		clone: function clone(object) {
151 			if (object instanceof Array) {
152 				return object.slice(0);
153 			} else if (typeof object == "object" && object !== null && !(object instanceof RegExp)) {
154 				return this.mixin(object, {});
155 			} else {
156 				return object;
157 			}
158 		}
159 		
160 	};
161 	
162 	
163 });

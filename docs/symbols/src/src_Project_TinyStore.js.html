1 define("TinyStore", ["Observable", "Tools"],
  2 /** 
  3  * @class
  4  * TinyStore creates a small NoSQL database
  5  * that can publish events on data add/change.
  6  */
  7  function TinyStore(Observable, Tools) {
  8 	
  9 	/**
 10 	 * Defines the tinyStore
 11 	 * @private
 12 	 * @param values
 13 	 * @returns
 14 	 */
 15 	return function TinyStoreConstructor($data) {
 16 		
 17 		/**
 18 		 * Where the data is stored
 19 		 * @private
 20 		 */
 21 		var _data = Tools.clone($data) || {},
 22 		
 23 		/**
 24 		 * The observable
 25 		 * @private
 26 		 */
 27 		_observable = new Observable(),
 28 		
 29 		/**
 30 		 * Gets the difference between two objects and notifies them
 31 		 * @private
 32 		 * @param previousData
 33 		 * @returns
 34 		 */
 35 		_notifyDiffs = function _notifyDiffs(previousData) {
 36 			var diffs = Tools.objectsDiffs(previousData, _data);
 37 			["updated",
 38 			 "deleted",
 39 			 "added"].forEach(function (value) {
 40 				 diffs[value].forEach(function (dataIndex) {
 41 						_observable.notify(value, dataIndex, _data[dataIndex]);
 42 				 });
 43 			});
 44 		};
 45 		
 46 		/**
 47 		 * Get the number of items in the store
 48 		 * @returns {Number} the number of items in the store
 49 		 */
 50 		this.getNbItems = function() {
 51 			return _data instanceof Array ? _data.length : Tools.count(_data);
 52 		};
 53 		
 54 		/**
 55 		 * Get a value from its index
 56 		 * @param {String} name the name of the index
 57 		 * @returns the value
 58 		 */
 59 		this.get = function get(name) {
 60 			return _data[name];
 61 		};
 62 		
 63 		/**
 64 		 * Checks if the store has a given value
 65 		 * @param {String} name the name of the index
 66 		 * @returns {Boolean} true if the value exists
 67 		 */
 68 		this.has = function has(name) {
 69 			return _data.hasOwnProperty(name);
 70 		};
 71 		
 72 		/**
 73 		 * Set a new value and overrides an existing one
 74 		 * @param {String} name the name of the value
 75 		 * @param value the value to assign
 76 		 * @returns true if value is set
 77 		 */
 78 		this.set = function set(name, value) {
 79 			var ante;
 80 			if (typeof name != "undefined") {
 81 				ante = this.has(name);
 82 				_data[name] = value;
 83 				if (ante) {
 84 					_observable.notify("updated", name, _data[name]);	
 85 				} else {
 86 					_observable.notify("added", name, _data[name]);
 87 				}
 88 				return true;
 89 			} else {
 90 				return false;
 91 			}
 92 		};
 93 		
 94 		/**
 95 		 * Delete value from its index
 96 		 * @param {String} name the name of the index from which to delete the value
 97 		 * @returns true if successfully deleted.
 98 		 */
 99 		this.del = function del(name) {
100 			if (this.has(name)) {
101 				this.alter("splice", name, 1) || delete _data[name];
102 				_observable.notify("deleted", name);
103 				return true;
104 			} else {
105 				return false;
106 			}
107 		};
108 		
109 		/**
110 		 * Alter the data be calling one of it's method
111 		 * When the modifications are done, it notifies on changes.
112 		 * @param {String} func the name of the method
113 		 * @returns the result of the method call
114 		 */
115 		this.alter = function alter(func) {
116 			var apply,
117 				previousData;
118 			
119 			if (_data[func]) {
120 				previousData = Tools.clone(_data);
121 				apply = _data[func].apply(_data, Array.prototype.slice.call(arguments, 1));
122 				_notifyDiffs(previousData);
123 				return apply;
124 			} else {
125 				return false;
126 			}
127 		};
128 		
129 		/**
130 		 * Watch the value's modifications
131 		 * @param {String} name the index of the value
132 		 * @param {Function} func the function to execute when the value changes
133 		 * @param {Object} scope the scope in which to execute the function
134 		 * @returns {Handler} the subscribe's handler to use to stop watching
135 		 */
136 		this.watch = function watch(name, func, scope) {
137 			return _observable.watch(name, func, scope);
138 		};
139 		
140 		/**
141 		 * Unwatch the value's modifications
142 		 * @param {Handler} handler the handler returned by the watch function
143 		 * @returns
144 		 */
145 		this.unwatch = function unwatch(handler) {
146 			return _observable.unwatch(handler);
147 		};
148 
149 		/**
150 		 * Loop through the data
151 		 * @param {Function} func the function to execute on each data
152 		 * @param {Object} scope the scope in wich to run the callback
153 		 */
154 		this.loop = function loop(func, scope) {
155 			Tools.loop(_data, func, scope);
156 		};
157 		
158 		/**
159 		 * Reset all data and get notifications on changes
160 		 * @param {Arra/Object} data the new data
161 		 * @returns {Boolean}
162 		 */
163 		this.reset = function reset(data) {
164 			if (data instanceof Object) {
165 				var previousData = Tools.clone(_data);
166 				_data = Tools.clone(data) || {};
167 				_notifyDiffs(previousData);
168 				return true;
169 			} else {
170 				return false;
171 			}
172 
173 		};
174 	};
175 });

1 define("TinyStore", ["Observable", "Tools"],
  2 /** 
  3  * @class
  4  * TinyStore creates a small NoSQL database
  5  * that can publish events on data add/change.
  6  */
  7  function TinyStore(Observable, Tools) {
  8 	
  9 	/**
 10 	 * Defines the tinyStore
 11 	 * @private
 12 	 * @param values
 13 	 * @returns
 14 	 */
 15 	function _TinyStore(values) {
 16 		
 17 		var _data = {}, 
 18 			_tools = Tools,
 19 			_observable = Observable.create();
 20 			
 21 		_tools.mixin(values, _data);
 22 		
 23 		/**
 24 		 * Get the number of items in the store
 25 		 * @returns {Int} the number of items in the store
 26 		 */
 27 		this.getNbItems = function() {
 28 			return _tools.count(_data);
 29 		};
 30 		
 31 
 32 		
 33 		/**
 34 		 * Get a value from its index
 35 		 * @param {String} name the name of the index
 36 		 * @returns the value
 37 		 */
 38 		this.get = function get(name) {
 39 			return _data[name];
 40 		};
 41 		
 42 		/**
 43 		 * Checks if the store has a given value
 44 		 * @param {String} name the name of the index
 45 		 * @returns {Boolean} true if the value exists
 46 		 */
 47 		this.has = function has(name) {
 48 			return _data.hasOwnProperty(name);
 49 		};
 50 		
 51 		/**
 52 		 * Set a new value and overrides an existing one
 53 		 * @param {String} name the name of the value
 54 		 * @param value the value to assign
 55 		 * @returns true if value is set
 56 		 */
 57 		this.set = function set(name, value) {
 58 			if (typeof name == "string") {
 59 				_data[name] = value;
 60 				_observable.notify(name, value);
 61 				return true;
 62 			} else {
 63 				return false;
 64 			}
 65 		};
 66 		
 67 		/**
 68 		 * Delete value from its index
 69 		 * @param {String} name the name of the index from which to delete the value
 70 		 * @returns true if successfully deleted.
 71 		 */
 72 		this.del = function del(name) {
 73 			if (this.has(name)) {
 74 				delete _data[name];
 75 				_observable.notify(name, _data[name]);
 76 				return true;
 77 			} else {
 78 				return false;
 79 			}
 80 		};
 81 		
 82 		/**
 83 		 * Watch the value's modifications
 84 		 * @param {String} name the index of the value
 85 		 * @param {Function} func the function to execute when the value changes
 86 		 * @param {Object} scope the scope in which to execute the function
 87 		 * @returns {Handler} the subscribe's handler to use to stop watching
 88 		 */
 89 		this.watch = function watch(name, func, scope) {
 90 			return _observable.watch(name, func, scope);
 91 		};
 92 		
 93 		/**
 94 		 * Unwatch the value's modifications
 95 		 * @param {Handler} handler the handler returned by the watch function
 96 		 * @returns
 97 		 */
 98 		this.unwatch = function unwatch(handler) {
 99 			return _observable.unwatch(handler);
100 		};
101 		
102 	}
103 	
104 	return { 
105 		/**
106 		 * Creates a new TinyStore
107 		 * @param {Object} values the values to initialize the store with
108 		 * @returns {Object} the TinyStore
109 		 */
110 		create: function create(values) {
111 			return new _TinyStore(values);
112 		}
113 	};
114 });

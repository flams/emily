  2 /**

  4 * Observable is an implementation of the Observer design pattern,

  7 * This service creates an Observable on which you can add subscribers.

 12 	/**
 13 	 * Returns a new Observable
 14 	 * @returns {Object} an Observable
 15 	 */

 17 		return new _Observable();
 18 	};

 20 	/**
 21 	 * Defines the Observable
 22 	 * @private
 23 	 * @returns {_Observable}
 24 	 */
 25 	function _Observable() {
 26 
 27 		/**
 28 		 * The list of topics
 29 		 * @private
 30 		 */
 31 		var _topics = {};
 32 		
 33 		/**
 34 		 * Add an observer
 35 		 * @param {String} topic the topic to observe
 36 		 * @param {Function} callback the callback to execute
 37 		 * @param {Object} scope the scope in which to execute the callback
 38 		 * @returns handler
 39 		 */
 40 		this.watch = function watch(topic, callback, scope) {
 41 			if (typeof topic == "string" && typeof callback == "function") {
 42 				var observers = _topics[topic] = _topics[topic] || [];
 43 			
 44 				observer = [callback, scope];
 45 				observers.push(observer);
 46 				return [topic,observers.indexOf(observer)];
 47 				
 48 			} else {
 49 				return false;
 50 			}
 51 		};
 52 		
 53 		/**
 54 		 * Remove an observer
 55 		 * @param {Handler} handler returned by the watch method
 56 		 * @returns {Boolean} true if there were subscribers
 57 		 */
 58 		this.unwatch = function unwatch(handler) {
 59 			var topic = handler[0], idx = handler[1];
 60 			if (_topics[topic] && _topics[topic][idx]) {
 61 				// set value to null so the indexes don't move.
 62 				_topics[topic][idx] = null;
 63 				return true;
 64 			} else {
 65 				return false;
 66 			}
 67 		};
 68 		
 69 		/**
 70 		 * Notifies observers that a topic has a new message
 71 		 * @param {String} topic the name of the topic to publish to
 72 		 * @param subject
 73 		 * @returns {Boolean} true if there was subscribers
 74 		 */
 75 		this.notify = function notify(topic, subject) {
 76 			
 77 			var observers = _topics[topic],
 78 				l;
 79 
 80 			if (observers) {
 81 				l = observers.length;
 82 				while (l--) {
 83 					observers[l] && observers[l][0].call(observers[l][1] || null, subject); 
 84 				}
 85 				return true;
 86 			} else {
 87 				return false;
 88 			}
 89 		},
 90 		
 91 		/**
 92 		 * Check if topic has the described observer
 93 		 * @param {Handler}
 94 		 * @returns {Boolean} true if exists
 95 		 */
 96 		this.hasObserver = function hasObserver(handler) {
 97 			return !!( handler && _topics[handler[0]] && _topics[handler[0]][handler[1]]);
 98 		};
 99 		
100 	}

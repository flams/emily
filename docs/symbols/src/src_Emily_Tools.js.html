1 define("Tools",
  2 /**
  3  * @class
  4  * Tools is a collection of tools 
  5  */
  6 function Tools (){
  7 
  8 	return {
  9 	    /**
 10 	     * For applications that don't run in a browser, window is not the global object.
 11 	     * This function returns the global object wherever the application runs.
 12 	     * @returns {Object} the global object
 13 	     */
 14 		getGlobal: function getGlobal() {
 15 	    	var func = function() {
 16 	    		return this;
 17 	    	};
 18 	    	return func.call(null);
 19 	    },
 20 		
 21 		/**
 22 		 * Mixes an object into another
 23 		 * @param {Object} destination object to mix values into
 24 		 * @param {Object} source object to get values from
 25 		 * @param {Boolean} optional, set to true to prevent overriding
 26 		 */
 27 	    mixin: function mixin(source, destination, dontOverride) {
 28 			var value;
 29 			for (value in source) {
 30 				if (source.hasOwnProperty(value)) {
 31 					if (!destination[value] || !dontOverride) {
 32 						destination[value] = source[value];	
 33 					}
 34 				}
 35 			}
 36 		},
 37 		
 38 		/**
 39 		 * Count the number of properties in an object
 40 		 * It doesn't look up in the prototype chain
 41 		 * @param {Object} object the object to count
 42 		 * @returns {Number}
 43 		 */
 44 		count: function count(object) {
 45 			var nbItems = 0,
 46 				v;
 47 			
 48 			for (v in object) {
 49 				if (object.hasOwnProperty(v)) {
 50 					nbItems++;
 51 				}
 52 			}
 53 			
 54 			return nbItems;
 55 		},
 56 		
 57 		/**
 58 		 * Compares the properties of two objects and returns true if they're the same
 59 		 * It's doesn't do it recursively
 60 		 * @param {Object} first object
 61 		 * @param {Object} second object
 62 		 * @returns {Boolean} true if the two objets have the same properties
 63 		 */
 64 		compareObjects: function compareObjects(object1, object2) {
 65 			var getOwnProperties = function (object) {
 66 				return Object.getOwnPropertyNames(object).sort().join("");
 67 			};
 68 			return getOwnProperties(object1) == getOwnProperties(object2);
 69 		},
 70 		
 71 		/**
 72 		 * Transform array-like objects to array, such as nodeLists or arguments
 73 		 * @param {Array-like object}
 74 		 * @returns {Array}
 75 		 */
 76 		toArray: function toArray(array) {
 77 			return Array.prototype.slice.call(array);
 78 		}
 79 	
 80 	};
 81 	
 82 });

  2 /**
  3  * @class

  6  function Tools(exports) {
  7 	
  8     
  9     exports.getGlobal = 
 10     /**
 11      * For applications that don't run in a browser, window is not the global object.
 12      * This function returns the global object wherever the application runs.
 13      * @returns {Object} the global object
 14      */
 15     function getGlobal() {
 16     	var func = function() {
 17     		return this;
 18     	};
 19     	return func.call(null);
 20     };
 21 	
 22 	exports.mixin = 
 23 	/**
 24 	 * Mixes an object into another
 25 	 * @param {Object} destination object to mix values into
 26 	 * @param {Object} source object to get values from
 27 	 * @param {Boolean} optional, set to true to prevent overriding
 28 	 */
 29 	function mixin(source, destination, dontOverride) {
 30 		var value;
 31 		for (value in source) {
 32 			if (source.hasOwnProperty(value)) {
 33 				if (!destination[value] || !dontOverride) {
 34 					destination[value] = source[value];	
 35 				}
 36 			}
 37 		}
 38 	};
 39 	
 40 	exports.count = 
 41 	/**
 42 	 * Count the number of properties in an object
 43 	 * It doesn't look up in the prototype chain
 44 	 * @param {Object} object the object to count
 45 	 * @returns {Number}
 46 	 */
 47 	function count(object) {
 48 		var nbItems = 0,
 49 			v;
 50 		
 51 		for (v in object) {
 52 			if (object.hasOwnProperty(v)) {
 53 				nbItems++;
 54 			}
 55 		}
 56 		
 57 		return nbItems;
 58 	};
 59 	
 60 });

 12 TestCase("ObservableWatch", {
 13 	
 14 	setUp: function () {
 15 		this.observable = Emily.Observable.create();
 16 		this.testTopic = "testTopic";
 17 	},
 18 	
 19 	"test should add observer": function () {
 20 		var func = function () {},
 21 			handler;
 22 		
 23 		assertFalse(this.observable.hasObserver(handler));
 24 		handler = this.observable.watch(this.testTopic, func);
 25 		
 26 		assertTrue(this.observable.hasObserver(handler));
 27 	},
 28 	
 29 	"test should add observer with scope": function () {
 30 		var func = function () {},
 31 			handler;
 32 		
 33 		assertFalse(this.observable.hasObserver(handler));
 34 		handler = this.observable.watch(this.testTopic, func, this);
 35 
 36 		assertTrue(this.observable.hasObserver(handler));
 37 	},
 38 	
 39 	"test should remove observer": function () {
 40 		var func = function () {
 41 				func.called = true;
 42 			},
 43 			handler;
 44 		
 45 		handler = this.observable.watch(this.testTopic, func);
 46 		
 47 		assertTrue(this.observable.hasObserver(handler));
 48 		this.observable.unwatch(handler);
 49 		assertFalse(this.observable.hasObserver(handler));
 50 		this.observable.notify(this.testTopic);
 51 		assertFalse(func.called === true);
 52 	},
 53 	
 54 	"test should return false if remove called with wrong handler": function () {
 55 		assertFalse(this.observable.unwatch("wrong", "handler"));
 56 	},
 57 	
 58 	"test should return false if wrong parameter count or type": function () {
 59 		assertFalse(this.observable.watch());
 60 		assertFalse(this.observable.watch("topic"));
 61 		assertFalse(this.observable.watch(function(){}, "topic"));
 62 		assertFalse(this.observable.watch("", {}));
 63 	}

 66 TestCase("ObservableNotify", {
 67 	
 68 	setUp : function () {
 69 		this.observable = Emily.Observable.create();
 70 		this.testTopic = "testTopic";
 71 	},
 72 	
 73 	"test should notify observer with no scope": function () {
 74 		var func = function () {
 75 				func.called  = true;
 76 			};
 77 
 78 		this.observable.watch(this.testTopic, func);
 79 		
 80 		assertUndefined(func.called);
 81 		this.observable.notify(this.testTopic);
 82 		assert(func.called);
 83 	},
 84 	
 85 	"test should notify observer with scope": function () {
 86 		var func = function () {
 87 				func.scope = this;
 88 			};
 89 		
 90 		this.observable.watch(this.testTopic, func, this);
 91 		
 92 		assertFalse(func.scope === this);
 93 		this.observable.notify(this.testTopic);
 94 		
 95 		assert(func.scope === this);
 96 	},
 97 	
 98 	"test should pass parameter": function () {
 99 		var func = function (p) {
100 			func.param = p;
101 		},
102 		post = {x:10};
103 		
104 		this.observable.watch(this.testTopic, func);
105 		
106 		assertFalse(func.param === post);
107 		this.observable.notify(this.testTopic, post);
108 		assert(func.param === post);
109 	},
110 	
111 	"test should return false if topic doesn't exist": function () {
112 		assertFalse(this.observable.notify("fakeTopic"));
113 	}
114 	
115 });

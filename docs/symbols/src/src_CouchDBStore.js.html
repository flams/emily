 11 /**
 12  * @class
 13  * CouchDBStore synchronises a Store with a CouchDB view or document
 14  * It subscribes to _changes to keep its data up to date.
 15  */
 16 function CouchDBStore(Store, StateMachine, Tools, Promise) {
 17 	
 18 	/**
 19 	 * Defines the CouchDBStore
 20 	 * @returns {CouchDBStoreConstructor}
 21 	 */
 22 	function CouchDBStoreConstructor() {
 23 		
 24 		/**
 25 		 * The name of the channel on which to run the requests
 26 		 * @private
 27 		 */
 28 		var _channel = "CouchDB",
 29 		
 30 		/**
 31 		 * The transport used to run the requests
 32 		 * @private
 33 		 */
 34 		_transport = null,
 35 		
 36 		/**
 37 		 * That will store the synchronization info
 38 		 * @private
 39 		 */
 40 		_syncInfo = {},
 41 		
 42 		/**
 43 		 * The promise that is returned by sync
 44 		 * It's resolved when entering listening state
 45 		 * It's rejected when no such document to sync to
 46 		 * @private
 47 		 */
 48 		_syncPromise = new Promise(),
 49 		
 50 		/**
 51 		 * All the actions performed by the couchDBStore
 52 		 * They'll feed the stateMachine
 53 		 * @private
 54 		 */
 55 		actions = {
 56 			
 57 			/**
 58 			 * Get a CouchDB view
 59 			 * @private
 60 			 */
 61 			getView: function () {
 62 
 63 				_syncInfo.query = _syncInfo.query || {};
 64 				_syncInfo.query.update_seq=true;
 65 				
 66 				_transport.request(_channel, {
 67 					method: "GET",
 68 					path: "/" + _syncInfo.database + "/_design/" + _syncInfo.design + "/" + "_view/" + _syncInfo.view,
 69 					query: _syncInfo.query
 70 				}, function (results) {
 71 					var json = JSON.parse(results);
 72 					if (!json.rows) {
 73 						throw new Error("CouchDBStore [" + _syncInfo.database + ", " + _syncInfo.design + ", " + _syncInfo.view + "].sync() failed: " + results);	
 74 					} else {
 75 						this.reset(json.rows);
 76 						_stateMachine.event("subscribeToViewChanges", json.update_seq);
 77 					}
 78 				}, this);
 79 			},
 80 			
 81 			/**
 82 			 * Get a CouchDB document
 83 			 * @private
 84 			 */
 85 			getDocument: function () { 
 86 				
 87 				_transport.request(_channel, {
 88 					method: "GET",
 89 					path: "/" + _syncInfo.database + "/" + _syncInfo.document,
 90 					query: _syncInfo.query
 91 				}, function (results) {
 92 					var json = JSON.parse(results);
 93 					if (json._id) {
 94 						this.reset(json);
 95 						_stateMachine.event("subscribeToDocumentChanges");	
 96 					} else {
 97 						_syncPromise.reject(this);
 98 					}
 99 				}, this);
100 			},
101 			
102 			/**
103 			 * Get a bulk of documents
104 			 * @private
105 			 */
106 			getBulkDocuments: function () {
107 				
108 				_syncInfo.query = _syncInfo.query || {};
109 				_syncInfo.query.include_docs = true;
110 				_syncInfo.query.update_seq = true;
111 				
112 				_transport.request(_channel, {
113 					method: "POST",
114 					path: "/" + _syncInfo.database + "/_all_docs",
115 					query: _syncInfo.query,
116 					headers: {
117 						"Content-Type": "application/json"
118 					},
119 					data: JSON.stringify({keys:_syncInfo.bulkDoc})
120 				}, function (results) {
121 					var json = JSON.parse(results);
122 					if (!json.rows) {
123 						throw new Error("CouchDBStore [" + _syncInfo.database + ", " + JSON.stringify(_syncInfo.bulkDoc) + "].sync() failed: " + results);	
124 					} else {
125 						this.reset(json.rows);
126 						_stateMachine.event("subscribeToBulkChanges", json.update_seq);
127 					}
128 				}, this);
129 				
130 			},
131 			
132 			/**
133 			 * Put a new document in CouchDB
134 			 * @private
135 			 */
136 			createDocument: function () {
137             	_transport.request(_channel, {
138             		method: "PUT",
139             		path: '/' + _syncInfo.database + '/' + _syncInfo.document,
140             		headers: {
141             			"Content-Type": "application/json"
142             		},
143             		data: this.toJSON()
144             	}, function () {
145             		_stateMachine.event("subscribeToDocumentChanges");
146             	});
147             },
148             
149             /**
150              * Subscribe to changes when synchronized with a view
151              * @param {Number} the update_seq given by getView, it'll be passed to since in the GET request
152              * @private
153              */
154             subscribeToViewChanges: function (update_seq) {
155             	
156             	Tools.mixin({
157 					feed: "continuous",
158 					heartbeat: 20000,
159 					since: update_seq
160 				}, _syncInfo.query);
161             	
162             	this.stopListening = _transport.listen(_channel, {
163 						path: "/" + _syncInfo.database + "/_changes",
164 						query: _syncInfo.query
165 					},
166 					function (changes) {
167 						// Should I test for this very special case (heartbeat?)
168 						// Or do I have to try catch for any invalid json?
169 						if (changes == "\n") {
170 							return false;
171 						}
172 						
173 						var json = JSON.parse(changes),
174 							action;
175 
176 						if (json.deleted) {
177 							action = "delete";
178 						} else if (json.changes[0].rev.search("1-") == 0) {
179 							action = "add";
180 						} else {
181 							action = "change";
182 						}
183 						_stateMachine.event(action, json.id);
184 					}, this);
185 			},
186 			
187 			/**
188 			 * Subscribe to changes when synchronized with a document
189 			 * @private
190 			 */
191 			subscribeToDocumentChanges: function () {
192 
193 				this.stopListening = _transport.listen(_channel, {
194 					path: "/" + _syncInfo.database + "/_changes",
195 					query: {
196 						 feed: "continuous",
197 						 heartbeat: 20000
198 						}
199 					},
200 				function (changes) {
201 					var json;
202 					// Should I test for this very special case (heartbeat?)
203 					// Or do I have to try catch for any invalid json?
204 					if (changes == "\n") {
205 						return false;
206 					}
207 					
208 					json = JSON.parse(changes);
209 					
210 					// The document is the modified document is the current one
211 					if (json.id == _syncInfo.document && 
212 						// And if it has a new revision
213 						json.changes.pop().rev != this.get("_rev")) {
214 						
215 						if (json.deleted) {
216 							_stateMachine.event("deleteDoc");
217 						} else {
218 							_stateMachine.event("updateDoc");	
219 						}
220 					 }
221 				}, this);
222 			},
223 			
224 			/**
225 			 * Subscribe to changes when synchronized with a bulk of documents
226 			 * @private
227 			 */
228 			subscribeToBulkChanges: function (update_seq) {
229 				Tools.mixin({
230 					feed: "continuous",
231 					heartbeat: 20000,
232 					since: update_seq
233 				}, _syncInfo.query);
234             	
235             	this.stopListening = _transport.listen(_channel, {
236 						path: "/" + _syncInfo.database + "/_changes",
237 						query: _syncInfo.query
238 					},
239 					function (changes) {
240 						var json;
241 						// Should I test for this very special case (heartbeat?)
242 						// Or do I have to try catch for any invalid json?
243 						if (changes == "\n") {
244 							return false;
245 						}
246 						
247 						var json = JSON.parse(changes),
248 							action;
249 						
250 						if (json.deleted) {
251 							action = "delete";
252 						} else {
253 							action = "bulkChange";
254 						}
255 						
256 						_stateMachine.event(action, json.id);
257 						
258 						
259 					}, this);
260 			},
261 			
262 			/**
263 			 * Update in the Store a document that was updated in CouchDB
264 			 * Get the whole view :(, then get the modified document and update it.
265 			 * I have no choice but to request the whole view and look for the document
266 			 * so I can also retrieve its position in the store (idx) and update the item.
267 			 * Maybe I've missed something
268 			 * @private
269 			 */
270 			updateDocInStore: function (id) {
271 				_transport.request(_channel,{
272 					method: "GET",
273 					path: '/'+_syncInfo.database+'/_design/'+_syncInfo.design+'/_view/'+_syncInfo.view,
274 					query: _syncInfo.query
275 				}, function (view) {
276 					var json = JSON.parse(view);
277 					
278 					json.rows.some(function (value, idx) {
279 						if (value.id == id) {
280 							this.set(idx, value);
281 						}
282 					}, this);
283 
284 					
285 				}, this);
286 				
287 			},
288 			
289 			/**
290 			 * Update in the Store a document that was updated in CouchDB
291 			 * Get all documents like the first sync request, then get the modified document and update it.
292 			 * I'm doing this because the document, the things I really care for (the rest is plumbing)
293 			 * is sometimes stored into value (view) into doc (bulk docs) or as is.
294 			 * @private
295 			 */
296 			updateBulkDocInStore: function (id) {
297 				_transport.request(_channel, {
298 					method: "POST",
299 					path: "/"+_syncInfo.database+"/_all_docs",
300 					query: _syncInfo.query
301 				}, function (bulk) {
302 					var json = JSON.parse(bulk);
303 					
304 					json.rows.some(function (value, idx) {
305 						if (value.id == id) {
306 							this.set(idx, value);
307 						}
308 					}, this);
309 					
310 				});
311 			},
312 			
313 			/**
314 			 * Remove from the Store a document that was removed in CouchDB
315 			 * @private
316 			 */
317 			removeDocInStore: function (id) {
318 				this.loop(function (value, idx) {
319 					if (value.id == id) {
320 						this.del(idx);
321 					}
322 				}, this);
323 			},
324 			
325 			/**
326 			 * Add in the Store a document that was added in CouchDB
327 			 * @private
328 			 */
329 			addDocInStore: function (id) {
330 				_transport.request(_channel,{
331 					method: "GET",
332 					path: '/'+_syncInfo.database+'/_design/'+_syncInfo.design+'/_view/'+_syncInfo.view,
333 					query: _syncInfo.query
334 				}, function (view) {
335 					var json = JSON.parse(view);
336 					
337 					json.rows.some(function (value, idx) {
338 						if (value.id == id) {
339 							this.alter("splice", idx, 0, value);	
340 						}
341 					}, this);
342 					
343 				}, this);
344 			},
345 			
346 			/**
347 			 * Update the document when synchronized with a document.
348 			 * This differs than updating a document in a View
349 			 * @private
350 			 */
351 			updateDoc: function () {
352 				_transport.request(_channel, {
353 					method: "GET",
354 					path: '/'+_syncInfo.database+'/' + _syncInfo.document
355 				}, function (doc) {
356 					this.reset(JSON.parse(doc));			
357 				}, this);
358 		    },
359 		    
360 		    /**
361 		     * Delete all document's properties
362 		     * @private
363 		     */
364 		    deleteDoc: function () {
365 		    	this.reset({});			
366 		    },
367 		    
368 		    /**
369 		     * Update a document in CouchDB through a PUT request
370 		     * @private
371 		     */
372 		    updateDatabase: function (callback) {
373 		    	_transport.request(_channel, {
374             		method: "PUT",
375             		path: '/' + _syncInfo.database + '/' + _syncInfo.document,
376             		headers: {
377             			"Content-Type": "application/json"
378             		},
379             		data: this.toJSON()
380             	}, function (response) {
381             		var json = JSON.parse(response);
382             		if (json.ok) {
383             			callback.resolve(json);
384             		} else {
385             			callback.reject(json);
386             		}
387             	});
388 		    },
389 		    
390 		    /**
391 		     * Update the database with bulk documents
392 		     * @private
393 		     */
394 		    updateDatabaseWithBulkDoc: function (callback) {
395 		    	
396 		    	var docs = [];
397 		    	this.loop(function (value) {
398 		    		docs.push(value.doc);
399 		    	});
400 		    	
401 		    	_transport.request(_channel, {
402 		    		method: "POST",
403 		    		path: "/" + _syncInfo.database + "/_bulk_docs",
404 		    		headers: {
405 		    			"Content-Type": "application/json"
406 		    		},
407 		    		data: JSON.stringify({"docs": docs})
408 		    	}, function (response) {
409             		callback.resolve(JSON.parse(response));
410             	});
411 		    },
412 		    
413 		    /**
414 		     * Remove a document from CouchDB through a DELETE request
415 		     * @private
416 		     */
417 		    removeFromDatabase: function () {
418 		    	_transport.request(_channel, {
419             		method: "DELETE",
420             		path: '/' + _syncInfo.database + '/' + _syncInfo.document,
421             		query: {
422             			rev: this.get("_rev")
423             		}
424             	});
425 		    },
426 		    
427 		    /**
428 		     * Resolve the promise
429 		     * @private
430 		     */
431 		    resolve: function () {
432             	  _syncPromise.resolve(this);
433              },
434              
435              /**
436               * The function call to unsync the store
437               * @private
438               */
439              unsync: function () {
440             	 this.stopListening();
441             	 delete this.stopListening;
442              }
443 		},
444 		
445 		/**
446 		 * The state machine
447 		 * @private
448 		 * it concentrates almost the whole logic.
449 		 */
450 		_stateMachine = new StateMachine("Unsynched", {
451 			"Unsynched": [
452 			    ["getView", actions.getView, this, "Synched"],
453 				["getDocument", actions.getDocument, this, "Synched"],
454 				["getBulkDocuments", actions.getBulkDocuments, this, "Synched"]
455 			 ],
456 						
457 			"Synched": [
458 			    ["updateDatabase", actions.createDocument, this],
459 			    ["subscribeToViewChanges", actions.subscribeToViewChanges, this, "Listening"],
460 				["subscribeToDocumentChanges", actions.subscribeToDocumentChanges, this, "Listening"],
461 				["subscribeToBulkChanges", actions.subscribeToBulkChanges, this, "Listening"],
462 				["unsync", function noop(){}, "Unsynched"]
463 			 ],
464 				
465 			"Listening": [
466 			    ["entry", actions.resolve, this],
467 			    ["change", actions.updateDocInStore, this],
468 			    ["bulkChange", actions.updateBulkDocInStore, this],
469 				["delete", actions.removeDocInStore, this],
470 				["add", actions.addDocInStore, this],
471 				["updateDoc", actions.updateDoc, this],
472 			    ["deleteDoc", actions.deleteDoc, this],
473 			    ["updateDatabase", actions.updateDatabase, this],
474 			    ["updateDatabaseWithBulkDoc", actions.updateDatabaseWithBulkDoc, this],
475 			    ["removeFromDatabase", actions.removeFromDatabase, this],
476 			    ["unsync", actions.unsync, this, "Unsynched"]
477 			   ]
478 			
479 		});
480 		
481 		/**
482 		 * Synchronize the store with a view
483 		 * @param {String} database the name of the database where to get...
484 		 * @param {String} ...design the design document, in which...
485 		 * @param {String} view ...the view is.
486 		 * @returns {Boolean}
487 		 */
488 		this.sync = function sync() {
489 			if (typeof arguments[0] == "string" && typeof arguments[1] == "string" && typeof arguments[2] == "string") {
490 				this.setSyncInfo(arguments[0], arguments[1], arguments[2], arguments[3]);
491 				_stateMachine.event("getView");
492 				return _syncPromise;
493 			} else if (typeof arguments[0] == "string" && typeof arguments[1] == "string" && typeof arguments[2] != "string") {
494 				this.setSyncInfo(arguments[0], arguments[1], arguments[2]);
495 				_stateMachine.event("getDocument");
496 				return _syncPromise;
497 			} else if (typeof arguments[0] == "string" && arguments[1] instanceof Array) {
498 				this.setSyncInfo(arguments[0], arguments[1], arguments[2]);
499 				_stateMachine.event("getBulkDocuments");
500 				return _syncPromise;
501 			}
502 			return false;
503 		};
504 		
505 		/**
506 		 * Set the synchronization information
507 		 * @private
508 		 * @returns {Boolean}
509 		 */
510 		this.setSyncInfo = function setSyncInfo() {
511 			if (typeof arguments[0] == "string" && typeof arguments[1] == "string" && typeof arguments[2] == "string") {
512 				_syncInfo["database"] = arguments[0];
513 				_syncInfo["design"] = arguments[1];
514 				_syncInfo["view"] = arguments[2];
515 				_syncInfo["query"] = arguments[3];
516 				return true;
517 			} else if (typeof arguments[0] == "string" && typeof arguments[1] == "string" && typeof arguments[2] != "string") {
518 				_syncInfo["database"] = arguments[0];
519 				_syncInfo["document"] = arguments[1];
520 				_syncInfo["query"] = arguments[2];
521 				return true;
522 			} else if (typeof arguments[0] == "string" && arguments[1] instanceof Array) {
523 				_syncInfo["database"] = arguments[0];
524 				_syncInfo["bulkDoc"] = arguments[1];
525 				_syncInfo["query"] = arguments[2];
526 				return true;
527 			}
528 			return false;
529 		};
530 		
531 		/**
532 		 * Get the synchronization information
533 		 * @private
534 		 * @returns
535 		 */
536 		this.getSyncInfo = function getSyncInfo() {
537 			return _syncInfo;
538 		};
539 		
540 		/**
541 		 * Unsync a store. Unsync must be called prior to resynchronization.
542 		 * That will prevent any unwanted resynchronization.
543 		 * Notice that previous data will still be available.
544 		 * @returns
545 		 */
546 		this.unsync = function unsync() {
547 			return _stateMachine.event("unsync");
548 		};
549 		
550 		/**
551 		 * Upload the document to the database
552 		 * Works for CouchDBStore that are synchronized with documents or bulk of documents.
553 		 * If synchronized with a bulk of documents, you can set the documents to delete _deleted property to true.
554 		 * No modification can be done on views.
555 		 * @returns true if upload called
556 		 */
557 		this.upload = function upload() {
558 			var promise = new Promise;
559 			if(_syncInfo.document) {
560 				_stateMachine.event("updateDatabase", promise);
561 				return promise;
562 			} else if (_syncInfo.bulkDoc) {
563 				_stateMachine.event("updateDatabaseWithBulkDoc", promise);
564 				return promise;
565 			} 
566 			return false;
567 		};
568 		
569 		/**
570 		 * Remove the document from the database
571 		 * @returns true if remove called
572 		 */
573 		this.remove = function remove() {
574 			if (_syncInfo.document) {
575 				return _stateMachine.event("removeFromDatabase");
576 			} 
577 			return false;
578 		};
579 		
580 		/**
581 		 * The transport object to use
582 		 * @param {Object} transport the transport object
583 		 * @returns {Boolean} true if 
584 		 */
585 		this.setTransport = function setTransport(transport) {
586 			if (transport && typeof transport.listen == "function" && typeof transport.request == "function") {
587 				_transport = transport;
588 				return true;
589 			} else {
590 				return false;
591 			}
592 		};
593 		
594 		/**
595 		 * Get the state machine
596 		 * Also only useful for debugging
597 		 * @private
598 		 * @returns {StateMachine} the state machine
599 		 */
600 		this.getStateMachine = function getStateMachine() {
601 			return _stateMachine;
602 		};
603 		
604 		/**
605 		 * Get the current transport
606 		 * Also only useful for debugging
607 		 * @private
608 		 * @returns {Object} the current transport
609 		 */
610 		this.getTransport = function getTransport() {
611 			return _transport;
612 		};
613 		
614 		/**
615 		 * The functions called by the stateMachine made available for testing purpose
616 		 * @private
617 		 */
618 		this.actions = actions;
619 
620 	};
621 	
622 	return function CouchDBStoreFactory() {
623 		CouchDBStoreConstructor.prototype = new Store;
624 		return new CouchDBStoreConstructor;
625 	};

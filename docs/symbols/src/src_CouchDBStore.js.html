1 define("CouchDBStore", 
  2 
  3 ["Store", "StateMachine", "Tools"], 
  4 
  5 /**
  6  * @class
  7  * CouchDBStore synchronises a Store with a CouchDB view
  8  * It subscribes to _changes to keep its data up to date.
  9  */
 10 function CouchDBStore(Store, StateMachine, Tools) {
 11 	
 12 	/**
 13 	 * Defines the CouchDBStore
 14 	 * @returns {CouchDBStoreConstructor}
 15 	 */
 16 	function CouchDBStoreConstructor() {
 17 		
 18 		/**
 19 		 * The name of the channel on which to run the requests
 20 		 * @private
 21 		 */
 22 		var _channel = "CouchDB",
 23 		
 24 		/**
 25 		 * The transport used to run the requests
 26 		 * @private
 27 		 */
 28 		_transport = null,
 29 		
 30 		/**
 31 		 * The database name
 32 		 * @private
 33 		 */
 34 		_database = null,
 35 		
 36 		/**
 37 		 * The name of the view
 38 		 * @private
 39 		 */
 40 		_view = null,
 41 		
 42 		/**
 43 		 * The name of the design document
 44 		 * @private
 45 		 */
 46 		_design = null,
 47 		
 48 		/**
 49 		 * An object to store info like update_sq
 50 		 * @private
 51 		 */
 52 		_dbInfo = {},
 53 		
 54 		/**
 55 		 * The state machine
 56 		 * @private
 57 		 * it concentrates almost the whole logic.
 58 		 * It can already be extended to handle reconnect for instance
 59 		 */
 60 		_stateMachine = new StateMachine("Unsynched", {
 61 			"Unsynched": [
 62 			              
 63 			 ["getView", function () {
 64 					_transport.request(_channel, {
 65 						method: "GET",
 66 						path: "/" + _database + "/_design/" + _design + "/" + "_view/" + _view +"?update_seq=true"
 67 					}, function (results) {
 68 						var json = JSON.parse(results);
 69 						_dbInfo = {
 70 								total_rows: json.total_rows,
 71 								update_seq: json.update_seq,
 72 								offset: json.offset
 73 						};
 74 						
 75 						this.reset(json.rows);
 76 						_stateMachine.event("subscribeToChanges", json.update_seq);
 77 					}, this);
 78 					
 79 				}, this, "Synched"]],
 80 						
 81 			"Synched": [
 82 			            
 83 			  ["subscribeToChanges", function (update_seq) {
 84 					_transport.listen(_channel, {
 85 						method: "GET",
 86 						path: "/" + _database + "/_changes?feed=continuous&heartbeat=20000&since="+update_seq
 87 					}, function (changes) {
 88 						
 89 						// Should I test for this very special case (heartbeat?)
 90 						// Or do I have to try catch for any invalid json?
 91 						if (changes == "\n") {
 92 							return false;
 93 						}
 94 						
 95 						var json = JSON.parse(changes),
 96 							action;
 97 
 98 						if (json.deleted) {
 99 							action = "delete";
100 						} else if (json.changes[0].rev[0] == "1") {
101 							action = "add";
102 						} else {
103 							action = "change";
104 						}
105 						_stateMachine.event(action, json.id);
106 					}, this);
107 				}, this, "Listening"]],
108 				
109 			"Listening": [
110 			              
111 			    ["change", function (id) {
112 					_transport.request(_channel,{
113 						method: "GET",
114 						path: '/'+_database+'/_design/'+_design+'/_view/'+_view
115 					}, function (view) {
116 						var json = JSON.parse(view);
117 						
118 						json.rows.some(function (value, idx) {
119 							if (value.id == id) {
120 								this.set(idx, value);
121 							}
122 						}, this);
123 
124 						
125 					}, this);
126 					
127 				}, this],
128 				
129 				["delete", function (id) {
130 					this.loop(function (value, idx) {
131 						if (value.id == id) {
132 							this.del(idx);
133 						}
134 					}, this);
135 				}, this],
136 				
137 				["add", function (id) {
138 					_transport.request(_channel,{
139 						method: "GET",
140 						path: '/'+_database+'/_design/'+_design+'/_view/'+_view
141 					}, function (view) {
142 						var json = JSON.parse(view);
143 						
144 						json.rows.some(function (value, idx) {
145 							if (value.id == id) {
146 								this.alter("splice", idx, 0, value);	
147 							}
148 						}, this);
149 						
150 					}, this);
151 				}, this]]
152 			
153 		});
154 		
155 		/**
156 		 * Synchronize the store with a view
157 		 * @param {String} database the name of the database where to get...
158 		 * @param {String} ...design the design document, in which...
159 		 * @param {String} view ...the view is.
160 		 * @returns {Boolean}
161 		 */
162 		this.sync = function sync(database, design, view) {
163 			
164 			if (typeof database == "string" && typeof design == "string" && typeof view == "string") {
165 				_database = database;
166 				_design = design;
167 				_view = view;
168 				_stateMachine.event("getView");
169 				return true;
170 			} else {
171 				return false;
172 			}
173 		};
174 		
175 		/**
176 		 * Get a specific info about the current view
177 		 * Should be used only for debugging
178 		 * @param {String} name (update_seq/offset/total_rows)
179 		 * Note: if you want to get the number of items, store.getNbItems() is the func for that
180 		 * @returns the info
181 		 */
182 		this.getDBInfo = function getDBInfo(name) {
183 			return _dbInfo[name];
184 		};
185 		
186 		/**
187 		 * The transport object to use
188 		 * @param {Object} transport the transport object
189 		 * @returns {Boolean} true if 
190 		 */
191 		this.setTransport = function setTransport(transport) {
192 			if (transport && typeof transport.listen == "function" && typeof transport.request == "function") {
193 				_transport = transport;
194 				return true;
195 			} else {
196 				return false;
197 			}
198 		};
199 		
200 		/**
201 		 * Get the current state machine's state
202 		 * Also only useful for debugging
203 		 * @returns {String} the name of the current state
204 		 */
205 		this.getState = function getState() {
206 			return _stateMachine.getCurrent();
207 		};
208 		
209 		/**
210 		 * Get the current transport
211 		 * Also only useful for debugging
212 		 * @returns {Object} the current transport
213 		 */
214 		this.getTransport = function getTransport() {
215 			return _transport;
216 		};
217 
218 	};
219 	
220 	return function CouchDBStoreFactory() {
221 		CouchDBStoreConstructor.prototype = new Store;
222 		return new CouchDBStoreConstructor;
223 	};
224 	
225 });

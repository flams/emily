 68 TestCase("TinyStoreWatch", {
 69 
 70 	setUp: function () {
 71 		this.tinyStore = Emily.require("TinyStore").create();
 72 	},
 73 
 74 	"test should be notified on set": function () {
 75 		var spy = sinon.spy();
 76 		this.tinyStore.watch("test", spy);
 77 		this.tinyStore.set("test");
 78 		
 79 		assertTrue(spy.called);
 80 		
 81 	},
 82 
 83 	"test should be notified on del": function () {
 84 		var spy = sinon.spy();
 85 		this.tinyStore.set("test");
 86 		
 87 		assertFalse(spy.called);
 88 		
 89 		this.tinyStore.watch("test", spy);		
 90 		this.tinyStore.del("test");
 91 		
 92 		assertTrue(spy.called);
 93 	},
 94 
 95 	"test should unwatch value": function () {
 96 		var spy = sinon.spy();
 97 		handler = this.tinyStore.watch("test", spy);
 98 		this.tinyStore.unwatch(handler);
 99 		
100 		this.tinyStore.set("test");
101 		this.tinyStore.del("test");
102 		
103 		assertFalse(spy.called);
104 	},
105 
106 	"test should receive new value on set": function () {
107 		var spy = sinon.spy(),
108 			value = {x:10};
109 
110 		this.tinyStore.watch("test", spy);
111 		this.tinyStore.set("test", value);
112 		
113 		assert(spy.calledWith(value));
114 		
115 		this.tinyStore.del("test");
116 		
117 		assert(spy.calledWith(undefined));
118 	},
119 
120 	"test should exec in scope on set": function () {
121 		var spy = sinon.spy(),
122 			thisObj = {};
123 
124 		this.tinyStore.watch("test", spy, thisObj);
125 		this.tinyStore.set("test");
126 		
127 		assertSame(thisObj, spy.thisValues[0]);
128 	},
129 
130 	"test should exec in scope on del": function () {
131 		var spy = sinon.spy(),
132 			thisObj = {};
133 
134 		this.tinyStore.set("test");
135 		this.tinyStore.watch("test", spy, thisObj);		
136 		this.tinyStore.del("test");
137 		
138 		assertSame(thisObj, spy.thisValues[0]);
139 	}

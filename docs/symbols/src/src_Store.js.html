  7 define("Store", ["Observable", "Tools"],
  8 /** 
  9  * @class
 10  * Store creates a small NoSQL database with observables
 11  * It can publish events on store/data changes
 12  */
 13  function Store(Observable, Tools) {
 14 	
 15 	/**
 16 	 * Defines the Store
 17 	 * @private
 18 	 * @param values
 19 	 * @returns
 20 	 */
 21 	return function StoreConstructor($data) {
 22 		
 23 		/**
 24 		 * Where the data is stored
 25 		 * @private
 26 		 */
 27 		var _data = Tools.clone($data) || {},
 28 		
 29 		/**
 30 		 * The observable
 31 		 * @private
 32 		 */
 33 		_storeObservable = new Observable(),
 34 		
 35 		_valueObservable = new Observable(),
 36 		
 37 		/**
 38 		 * Gets the difference between two objects and notifies them
 39 		 * @private
 40 		 * @param previousData
 41 		 * @returns
 42 		 */
 43 		_notifyDiffs = function _notifyDiffs(previousData) {
 44 			var diffs = Tools.objectsDiffs(previousData, _data);
 45 			["updated",
 46 			 "deleted",
 47 			 "added"].forEach(function (value) {
 48 				 diffs[value].forEach(function (dataIndex) {
 49 						_storeObservable.notify(value, dataIndex, _data[dataIndex]);
 50 						_valueObservable.notify(dataIndex, _data[dataIndex], value);
 51 				 });
 52 			});
 53 		};
 54 		
 55 		/**
 56 		 * Get the number of items in the store
 57 		 * @returns {Number} the number of items in the store
 58 		 */
 59 		this.getNbItems = function() {
 60 			return _data instanceof Array ? _data.length : Tools.count(_data);
 61 		};
 62 		
 63 		/**
 64 		 * Get a value from its index
 65 		 * @param {String} name the name of the index
 66 		 * @returns the value
 67 		 */
 68 		this.get = function get(name) {
 69 			return _data[name];
 70 		};
 71 		
 72 		/**
 73 		 * Checks if the store has a given value
 74 		 * @param {String} name the name of the index
 75 		 * @returns {Boolean} true if the value exists
 76 		 */
 77 		this.has = function has(name) {
 78 			return _data.hasOwnProperty(name);
 79 		};
 80 		
 81 		/**
 82 		 * Set a new value and overrides an existing one
 83 		 * @param {String} name the name of the index
 84 		 * @param value the value to assign
 85 		 * @returns true if value is set
 86 		 */
 87 		this.set = function set(name, value) {
 88 			var ante,
 89 				action;
 90 			
 91 			if (typeof name != "undefined") {
 92 				ante = this.has(name);
 93 				_data[name] = value;
 94 				action = ante ? "updated" : "added";
 95 				_storeObservable.notify(action, name, _data[name]);	
 96 				_valueObservable.notify(name, _data[name], action);
 97 				return true;
 98 			} else {
 99 				return false;
100 			}
101 		};
102 		
103 		/**
104 		 * Update the property of an item.
105 		 * @param {String} name the name of the index
106 		 * @param {String} property the property to modify.
107 		 * @param value the value to assign
108 		 * @returns false if the Store has no name index
109 		 */
110 		this.update = function update(name, property, value) {
111 			var item;
112 			if (this.has(name)) {
113 				item = this.get(name);
114 				Tools.setNestedProperty(item, property, value);
115 				_storeObservable.notify("updated", property, value);
116 				_valueObservable.notify(name, item, "updated");
117 				return true;
118 			} else {
119 				return false;
120 			}
121 		};
122 		
123 		/**
124 		 * Delete value from its index
125 		 * @param {String} name the name of the index from which to delete the value
126 		 * @returns true if successfully deleted.
127 		 */
128 		this.del = function del(name) {
129 			if (this.has(name)) {
130 				if (!this.alter("splice", name, 1)) {
131 					delete _data[name];
132 					_storeObservable.notify("deleted", name);
133 					_valueObservable.notify(name, _data[name], "deleted");
134 				}
135 				return true;
136 			} else {
137 				return false;
138 			}
139 		};
140 		
141 		/**
142 		 * Delete multiple indexes. Prefer this one over multiple del calls.
143 		 * @param {Array} 
144 		 * @returns false if param is not an array.
145 		 */
146 		this.delAll = function delAll(indexes) {
147 			if (indexes instanceof Array) {
148 				// Indexes must be removed from the greatest to the lowest
149 				// To avoid trying to remove indexes that don't exist.
150 				// i.e: given [0, 1, 2], remove 1, then 2, 2 doesn't exist anymore
151 				indexes.sort(Tools.compareNumbers).reverse().forEach(this.del, this);
152 				return true;
153 			} else {
154 				return false;
155 			}
156 		};
157 		
158 		/**
159 		 * Alter the data be calling one of it's method
160 		 * When the modifications are done, it notifies on changes.
161 		 * @param {String} func the name of the method
162 		 * @returns the result of the method call
163 		 */
164 		this.alter = function alter(func) {
165 			var apply,
166 				previousData;
167 			
168 			if (_data[func]) {
169 				previousData = Tools.clone(_data);
170 				apply = _data[func].apply(_data, Array.prototype.slice.call(arguments, 1));
171 				_notifyDiffs(previousData);
172 				return apply;
173 			} else {
174 				return false;
175 			}
176 		};
177 		
178 		/**
179 		 * Watch the store's modifications
180 		 * @param {String} added/updated/deleted
181 		 * @param {Function} func the function to execute
182 		 * @param {Object} scope the scope in which to execute the function
183 		 * @returns {Handler} the subscribe's handler to use to stop watching
184 		 */
185 		this.watch = function watch(name, func, scope) {
186 			return _storeObservable.watch(name, func, scope);
187 		};
188 		
189 		/**
190 		 * Unwatch the store modifications
191 		 * @param {Handler} handler the handler returned by the watch function
192 		 * @returns
193 		 */
194 		this.unwatch = function unwatch(handler) {
195 			return _storeObservable.unwatch(handler);
196 		};
197 		
198 		/**
199 		 * Get the observable used for watching store's modifications
200 		 * Should be used only for debugging
201 		 * @returns {Observable} the Observable
202 		 */
203 		this.getStoreObservable = function getStoreObservable() {
204 			return _storeObservable;
205 		};
206 		
207 		/**
208 		 * Watch a value's modifications
209 		 * @param {String} name the name of the value to watch for
210 		 * @param {Function} func the function to execute
211 		 * @param {Object} scope the scope in which to execute the function
212 		 * @returns true if unwatched
213 		 */
214 		this.watchValue = function watchValue(name, func, scope) {
215 			return _valueObservable.watch(name, func, scope);
216 		};
217 		
218 		/**
219 		 * Unwatch the value's modifications
220 		 * @param {Handler} handler the handler returned by the watchValue function
221 		 * @returns true if unwatched
222 		 */
223 		this.unwatchValue = function unwatchValue(handler) {
224 			return _valueObservable.unwatch(handler);
225 		};
226 		
227 		/**
228 		 * Get the observable used for watching value's modifications
229 		 * Should be used only for debugging
230 		 * @returns {Observable} the Observable
231 		 */
232 		this.getValueObservable = function getValueObservable() {
233 			return _valueObservable;
234 		};
235 
236 		/**
237 		 * Loop through the data
238 		 * @param {Function} func the function to execute on each data
239 		 * @param {Object} scope the scope in wich to run the callback
240 		 */
241 		this.loop = function loop(func, scope) {
242 			Tools.loop(_data, func, scope);
243 		};
244 		
245 		/**
246 		 * Reset all data and get notifications on changes
247 		 * @param {Arra/Object} data the new data
248 		 * @returns {Boolean}
249 		 */
250 		this.reset = function reset(data) {
251 			if (data instanceof Object) {
252 				var previousData = Tools.clone(_data);
253 				_data = Tools.clone(data) || {};
254 				_notifyDiffs(previousData);
255 				return true;
256 			} else {
257 				return false;
258 			}
259 
260 		};
261 		
262 		/**
263 		 * Dumps a JSON version of all the data
264 		 * @returns {JSON}
265 		 */
266 		this.toJSON = function toJSON() {
267 			return JSON.stringify(_data);
268 		};
269 	};
270 });

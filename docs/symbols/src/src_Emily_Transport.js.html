1 Emily.declare("Transport",
  2 /** 
  3  * @class
  4  * Transport allows for client-server eventing.
  5  * It's based on socket.io.
  6  */
  7 function Transport(exports, API) {
  8 	
  9 	exports.create = 
 10 	/**
 11 	 * Creates a new Transport
 12 	 * @param {Url} url the url to connect Transport to
 13 	 * @returns {Object} new Transport
 14 	 */
 15 	function create(url) {
 16 		return new _Transport(url).connect(url);
 17 	};
 18 	
 19 	/**
 20 	 * Defines the Transport
 21 	 * @private
 22 	 * @param {url} $url the url to connect Transport to
 23 	 * @returns
 24 	 */
 25 	function _Transport($url) {
 26 		
 27 		/**
 28 		 * @private
 29 		 * The socket.io's socket
 30 		 */
 31 		var _socket,
 32 		
 33 		/**
 34 		 * @private
 35 		 * The socket.io globally defined module
 36 		 */
 37 		_io = API.require("io");
 38 		
 39 		/**
 40 		 * Connect Transport to an url
 41 		 * @param {Url} url the url to connect Transport to
 42 		 */
 43 		this.connect = function connect(url) {
 44 			_socket = _io.connect(url);
 45 			return this;
 46 		},
 47 		
 48 		/**
 49 		 * Get the socket, for debugging purpose
 50 		 * @returns {Object} the socket
 51 		 */
 52 		this.getSocket = function getSocket() {
 53 			return _socket;
 54 		},
 55 		
 56 		/**
 57 		 * Subscribe to a socket event
 58 		 * @param {String} event the name of the event
 59 		 * @param {Function} func the function to execute when the event fires
 60 		 */
 61 		this.on = function on(event, func) {
 62 			_socket.on(event, func);
 63 		},
 64 		
 65 		/**
 66 		 * Publish an event on the socket
 67 		 * @param {String} event the event to publish
 68 		 * @param data
 69 		 */
 70 		this.emit = function emit(event, data) {
 71 			_socket.emit(event, data);
 72 		};
 73 	};
 74 });

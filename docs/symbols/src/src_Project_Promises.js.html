1 define("Promise", 
  2 
  3 ["StateMachine", "Observable", "Tools"],
  4 
  5 /**
  6  * @class
  7  * This is an attempt to implement promises.
  8  * Not sure it's correct at the moment
  9  */
 10 function Promise(StateMachine, Observable, Tools) {
 11 	
 12 	/**
 13 	 * Defines the promise
 14 	 */
 15 	return function PromiseConstructor($func, $scope) {
 16 		
 17 		/**
 18 		 * The function to resolve
 19 		 * @private
 20 		 */
 21 		var _toResolve = null,
 22 		
 23 		/**
 24 		 * The scope in which to run the function to resolve
 25 		 * @private
 26 		 */
 27 		_scope = null,
 28 		
 29 		/**
 30 		 * The value to pass to success callbacks
 31 		 * @private
 32 		 */
 33 		_success = null,
 34 		
 35 		/**
 36 		 * The value to pass to errbacks
 37 		 * @private
 38 		 */
 39 		_failed = null,
 40 		
 41 		/**
 42 		 * The observable to notifiy the result
 43 		 * @private
 44 		 */
 45 		_observable = new Observable();
 46 		
 47 		/**
 48 		 * The state machine
 49 		 * 3 states : unresolved, resolved, failed
 50 		 * @private
 51 		 */
 52 		_stateMachine = new StateMachine("Unresolved", {
 53 			"Unresolved": [["resolve", function () {
 54 					_toResolve.call(_scope, function () {
 55 						return _stateMachine.event.apply(_stateMachine, Tools.toArray(arguments));
 56 					});
 57 				}],
 58 				["success", function (args) {
 59 					_success = args;
 60 					_observable.notify("success", args);
 61 				}, "Resolved"],
 62 				["fail", function (args) {
 63 					_failed = args;
 64 					_observable.notify("fail", args);
 65 				}, "Failed"],
 66 				["addThen", function (type, func, scope) {
 67 					_observable.watch(type, func, scope);
 68 				}]],
 69 			"Resolved": [["addThen", function (type, func, scope) {
 70 				func.call(scope, type == "success" ? _success : _failed);
 71 			}]],
 72 			"Failed": [["addThen", function (type, func, scope) {
 73 				func.call(scope, type == "success" ? _success : _failed);
 74 			}]]
 75 		});
 76 		
 77 		/**
 78 		 * The callbacks and errbacks to call after resolution.
 79 		 * @param {Function} the first parameter is a success function, it can be followed by a scope in which to run it
 80 		 * @param {Function} the second, or third parameter is an errback, it can also be followed by a scope
 81 		 * @examples:
 82 		 * 
 83 		 * then(callback)
 84 		 * then(callback, scope, errback, scope)
 85 		 * then(callback, errback)
 86 		 * then(callback, errback, scope)
 87 		 * 
 88 		 */	
 89 		this.then = function then() {
 90 
 91 			if (arguments[0] instanceof Function) {
 92 				if (arguments[1] instanceof Function) {
 93 					_stateMachine.event("addThen", "success", arguments[1]);
 94 				} else {
 95 					_stateMachine.event("addThen", "success", arguments[0], arguments[1]);
 96 				}
 97 			}
 98 			
 99 			if (arguments[1] instanceof Function) {
100 				_stateMachine.event("addThen", "fail", arguments[1], arguments[2]);
101 			} 
102 			
103 			if (arguments[2] instanceof Function) {
104 				_stateMachine.event("addThen", "fail", arguments[2], arguments[3]);
105 			}
106 
107 		};
108 		
109 		/**
110 		 * Launch the promise resolution
111 		 * @returns {Boolean} true if the resolution start
112 		 */
113 		this.resolve = function resolve() {
114 			return !!(_toResolve && _stateMachine.event("resolve"));
115 		};
116 		
117 		/**
118 		 * Get the current state machine's state
119 		 * Only useful for debugging
120 		 * @returns {String}
121 		 */
122 		this.getState = function getState() {
123 			return _stateMachine.getCurrent();
124 		};
125 		
126 		if ($func instanceof Function) {
127 			_toResolve = $func;
128 			_scope = $scope;
129 		}
130 		
131 	};
132 	
133 });

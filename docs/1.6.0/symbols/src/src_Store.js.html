  7 define(["Observable", "Tools"],
  8 /**
  9  * @class
 10  * Store creates an observable structure based on a key/values object
 11  * or on an array
 12  */
 13  function Store(Observable, Tools) {
 14 
 15     /**
 16      * Defines the Store
 17      * @param {Array/Object} the data to initialize the store with
 18      * @returns
 19      */
 20     return function StoreConstructor($data) {
 21 
 22         /**
 23          * Where the data is stored
 24          * @private
 25          */
 26         var _data = Tools.clone($data) || {},
 27 
 28         /**
 29          * The observable for publishing changes on the store iself
 30          * @private
 31          */
 32         _storeObservable = new Observable(),
 33 
 34         /**
 35          * The observable for publishing changes on a value
 36          * @private
 37          */
 38         _valueObservable = new Observable(),
 39 
 40 		/**
 41          * Saves the handles for the subscriptions of the computed properties
 42          * @private
 43          */
 44         _computed = [],
 45 
 46         /**
 47          * Gets the difference between two objects and notifies them
 48          * @private
 49          * @param {Object} previousData
 50          */
 51         _notifyDiffs = function _notifyDiffs(previousData) {
 52             var diffs = Tools.objectsDiffs(previousData, _data);
 53             ["updated",
 54              "deleted",
 55              "added"].forEach(function (value) {
 56                  diffs[value].forEach(function (dataIndex) {
 57                         _storeObservable.notify(value, dataIndex, _data[dataIndex]);
 58                         _valueObservable.notify(dataIndex, _data[dataIndex], value);
 59                  });
 60             });
 61         };
 62 
 63         /**
 64          * Get the number of items in the store
 65          * @returns {Number} the number of items in the store
 66          */
 67         this.getNbItems = function() {
 68             return _data instanceof Array ? _data.length : Tools.count(_data);
 69         };
 70 
 71         /**
 72          * Count is an alias for getNbItems
 73          * @returns {Number} the number of items in the store
 74          */
 75         this.count = this.getNbItems;
 76 
 77         /**
 78          * Get a value from its index
 79          * @param {String} name the name of the index
 80          * @returns the value
 81          */
 82         this.get = function get(name) {
 83             return _data[name];
 84         };
 85 
 86         /**
 87          * Checks if the store has a given value
 88          * @param {String} name the name of the index
 89          * @returns {Boolean} true if the value exists
 90          */
 91         this.has = function has(name) {
 92             return _data.hasOwnProperty(name);
 93         };
 94 
 95         /**
 96          * Set a new value and overrides an existing one
 97          * @param {String} name the name of the index
 98          * @param value the value to assign
 99          * @returns true if value is set
100          */
101         this.set = function set(name, value) {
102             var hasPrevious,
103                 previousValue,
104                 action;
105 
106             if (typeof name != "undefined") {
107                 hasPrevious = this.has(name);
108                 previousValue = this.get(name);
109                 _data[name] = value;
110                 action = hasPrevious ? "updated" : "added";
111                 _storeObservable.notify(action, name, _data[name], previousValue);
112                 _valueObservable.notify(name, _data[name], action, previousValue);
113                 return true;
114             } else {
115                 return false;
116             }
117         };
118 
119         /**
120          * Update the property of an item.
121          * @param {String} name the name of the index
122          * @param {String} property the property to modify.
123          * @param value the value to assign
124          * @returns false if the Store has no name index
125          */
126         this.update = function update(name, property, value) {
127             var item;
128             if (this.has(name)) {
129                 item = this.get(name);
130                 Tools.setNestedProperty(item, property, value);
131                 _storeObservable.notify("updated", property, value);
132                 _valueObservable.notify(name, item, "updated");
133                 return true;
134             } else {
135                 return false;
136             }
137         };
138 
139         /**
140          * Delete value from its index
141          * @param {String} name the name of the index from which to delete the value
142          * @returns true if successfully deleted.
143          */
144         this.del = function del(name) {
145             if (this.has(name)) {
146                 if (!this.alter("splice", name, 1)) {
147                     delete _data[name];
148                     _storeObservable.notify("deleted", name);
149                     _valueObservable.notify(name, _data[name], "deleted");
150                 }
151                 return true;
152             } else {
153                 return false;
154             }
155         };
156 
157         /**
158          * Delete multiple indexes. Prefer this one over multiple del calls.
159          * @param {Array}
160          * @returns false if param is not an array.
161          */
162         this.delAll = function delAll(indexes) {
163             if (indexes instanceof Array) {
164                 // Indexes must be removed from the greatest to the lowest
165                 // To avoid trying to remove indexes that don't exist.
166                 // i.e: given [0, 1, 2], remove 1, then 2, 2 doesn't exist anymore
167                 indexes.sort(Tools.compareNumbers)
168                     .reverse()
169                     .forEach(this.del, this);
170                 return true;
171             } else {
172                 return false;
173             }
174         };
175 
176         /**
177          * Alter the data by calling one of it's method
178          * When the modifications are done, it notifies on changes.
179          * If the function called doesn't alter the data, consider using proxy instead
180          * which is much, much faster.
181          * @param {String} func the name of the method
182          * @params {*} any number of params to be given to the func
183          * @returns the result of the method call
184          */
185         this.alter = function alter(func) {
186             var apply,
187                 previousData;
188 
189             if (_data[func]) {
190                 previousData = Tools.clone(_data);
191                 apply = this.proxy.apply(this, arguments);
192                 _notifyDiffs(previousData);
193                 return apply;
194             } else {
195                 return false;
196             }
197         };
198 
199         /**
200          * Proxy is similar to alter but doesn't trigger events.
201          * It's preferable to call proxy for functions that don't
202          * update the interal data source, like slice or filter.
203          * @param {String} func the name of the method
204          * @params {*} any number of params to be given to the func
205          * @returns the result of the method call
206          */
207         this.proxy = function proxy(func) {
208         	if (_data[func]) {
209         		return _data[func].apply(_data, Array.prototype.slice.call(arguments, 1));
210         	} else {
211         		return false;
212         	}
213         };
214 
215         /**
216          * Watch the store's modifications
217          * @param {String} added/updated/deleted
218          * @param {Function} func the function to execute
219          * @param {Object} scope the scope in which to execute the function
220          * @returns {Handle} the subscribe's handler to use to stop watching
221          */
222         this.watch = function watch(name, func, scope) {
223             return _storeObservable.watch(name, func, scope);
224         };
225 
226         /**
227          * Unwatch the store modifications
228          * @param {Handle} handle the handler returned by the watch function
229          * @returns
230          */
231         this.unwatch = function unwatch(handle) {
232             return _storeObservable.unwatch(handle);
233         };
234 
235         /**
236          * Get the observable used for watching store's modifications
237          * Should be used only for debugging
238          * @returns {Observable} the Observable
239          */
240         this.getStoreObservable = function getStoreObservable() {
241             return _storeObservable;
242         };
243 
244         /**
245          * Watch a value's modifications
246          * @param {String} name the name of the value to watch for
247          * @param {Function} func the function to execute
248          * @param {Object} scope the scope in which to execute the function
249          * @returns handler to pass to unwatchValue
250          */
251         this.watchValue = function watchValue(name, func, scope) {
252             return _valueObservable.watch(name, func, scope);
253         };
254 
255         /**
256          * Unwatch the value's modifications
257          * @param {Handler} handler the handler returned by the watchValue function
258          * @private
259          * @returns true if unwatched
260          */
261         this.unwatchValue = function unwatchValue(handler) {
262             return _valueObservable.unwatch(handler);
263         };
264 
265         /**
266          * Get the observable used for watching value's modifications
267          * Should be used only for debugging
268          * @private
269          * @returns {Observable} the Observable
270          */
271         this.getValueObservable = function getValueObservable() {
272             return _valueObservable;
273         };
274 
275         /**
276          * Loop through the data
277          * @param {Function} func the function to execute on each data
278          * @param {Object} scope the scope in wich to run the callback
279          */
280         this.loop = function loop(func, scope) {
281             Tools.loop(_data, func, scope);
282         };
283 
284         /**
285          * Reset all data and get notifications on changes
286          * @param {Arra/Object} data the new data
287          * @returns {Boolean}
288          */
289         this.reset = function reset(data) {
290             if (data instanceof Object) {
291                 var previousData = Tools.clone(_data);
292                 _data = Tools.clone(data) || {};
293                 _notifyDiffs(previousData);
294                 return true;
295             } else {
296                 return false;
297             }
298 
299         };
300 
301         /**
302          * Compute a new property from other properties.
303          * The computed property will look exactly similar to any none
304          * computed property, it can be watched upon.
305          * @param {String} name the name of the computed property
306          * @param {Array} computeFrom a list of properties to compute from
307          * @param {Function} callback the callback to compute the property
308          * @param {Object} scope the scope in which to execute the callback
309          * @returns {Boolean} false if wrong params given to the function
310          */
311         this.compute = function compute(name, computeFrom, callback, scope) {
312         	var args = [];
313 
314         	if (typeof name == "string" &&
315         		typeof computeFrom == "object" &&
316         		typeof callback == "function" &&
317         		!this.isCompute(name)) {
318 
319         		_computed[name] = [];
320 
321         		Tools.loop(computeFrom, function (property) {
322         			_computed[name].push(this.watchValue(property, function () {
323         				this.set(name, callback.call(scope));
324         			}, this));
325         		}, this);
326 
327         		this.set(name, callback.call(scope));
328         		return true;
329         	} else {
330         		return false;
331         	}
332         };
333 
334         /**
335          * Remove a computed property
336          * @param {String} name the name of the computed to remove
337          * @returns {Boolean} true if the property is removed
338          */
339         this.removeCompute = function removeCompute(name) {
340         	if (this.isCompute(name)) {
341         		Tools.loop(_computed[name], function (handle) {
342         			this.unwatchValue(handle);
343         		}, this);
344         		this.del(name);
345         		return true;
346         	} else {
347         		return false;
348         	}
349         };
350 
351         /**
352          * Tells if a property is a computed property
353          * @param {String} name the name of the property to test
354          * @returns {Boolean} true if it's a computed property
355          */
356         this.isCompute = function isCompute(name) {
357         	return !!_computed[name];
358         };
359 
360         /**
361          * Returns a JSON version of the data
362          * Use dump if you want all the data as a plain js object
363          * @returns {String} the JSON
364          */
365         this.toJSON = function toJSON() {
366             return JSON.stringify(_data);
367         };
368 
369         /**
370          * Returns the store's data
371          * @returns {Object} the data
372          */
373         this.dump = function dump() {
374             return _data;
375         };
376     };
377 });

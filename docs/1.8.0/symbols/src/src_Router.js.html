  7 define(["Observable", "Store", "Tools"],
  8 
  9 /**
 10  * @class
 11  * Routing allows for navigating in an application by defining routes.
 12  */
 13 function Router(Observable, Store, Tools) {
 14 
 15     "use strict";
 16 
 17     return function RouterConstructor() {
 18 
 19         /**
 20          * The routes observable (the applications use it)
 21          * @private
 22          */
 23         var _routes = new Observable(),
 24 
 25         /**
 26          * The events observable (used by Routing)
 27          * @private
 28          */
 29         _events = new Observable(),
 30 
 31         /**
 32          * The routing history
 33          * @private
 34          */
 35         _history = new Store([]),
 36 
 37         /**
 38          * For navigating through the history, remembers the current position
 39          * @private
 40          */
 41         _currentPos = -1,
 42 
 43         /**
 44          * The depth of the history
 45          * @private
 46          */
 47         _maxHistory = 10;
 48 
 49         /**
 50          * Only for debugging
 51          * @private
 52          */
 53         this.getRoutesObservable = function getRoutesObservable() {
 54             return _routes;
 55         };
 56 
 57         /**
 58          * Only for debugging
 59          * @private
 60          */
 61         this.getEventsObservable = function getEventsObservable() {
 62             return _events;
 63         };
 64 
 65         /**
 66          * Set the maximum length of history
 67          * As the user navigates through the application, the
 68          * routeur keeps track of the history. Set the depth of the history
 69          * depending on your need and the amount of memory that you can allocate it
 70          * @param {Number} maxHistory the depth of history
 71          * @returns {Boolean} true if maxHistory is equal or greater than 0
 72          */
 73         this.setMaxHistory = function setMaxHistory(maxHistory) {
 74             if (maxHistory >= 0) {
 75                 _maxHistory = maxHistory;
 76                 return true;
 77             } else {
 78                 return false;
 79             }
 80 
 81         };
 82 
 83         /**
 84          * Get the current max history setting
 85          * @returns {Number} the depth of history
 86          */
 87         this.getMaxHistory = function getMaxHistory() {
 88             return _maxHistory;
 89         };
 90 
 91         /**
 92          * Set a new route
 93          * @param {String} route the name of the route
 94          * @param {Function} func the function to be execute when navigating to the route
 95          * @param {Object} scope the scope in which to execute the function
 96          * @returns a handle to remove the route
 97          */
 98         this.set = function set() {
 99             return _routes.watch.apply(_routes, arguments);
100         };
101 
102         /**
103          * Remove a route
104          * @param {Object} handle the handle provided by the set method
105          * @returns true if successfully removed
106          */
107         this.unset = function unset(handle) {
108             return _routes.unwatch(handle);
109         };
110 
111         /**
112          * Navigate to a route
113          * @param {String} route the route to navigate to
114          * @param {*} *params
115          * @returns
116          */
117         this.navigate = function get(route, params) {
118             if (this.load.apply(this, arguments)) {
119                 // Before adding a new route to the history, we must clear the forward history
120                 _history.proxy("splice", _currentPos +1, _history.count());
121                 _history.proxy("push", Tools.toArray(arguments));
122                 this.ensureMaxHistory(_history);
123                 _currentPos = _history.count() -1;
124                 return true;
125             } else {
126                 return false;
127             }
128 
129         };
130 
131         /**
132          * Ensure that history doesn't grow bigger than the max history setting
133          * @param {Store} history the history store
134          * @private
135          */
136         this.ensureMaxHistory = function ensureMaxHistory(history) {
137             var count = history.count(),
138                 max = this.getMaxHistory(),
139                 excess = count - max;
140 
141             if (excess > 0) {
142                 history.proxy("splice", 0, excess);
143             }
144         };
145 
146         /**
147          * Actually loads the route
148          * @private
149          */
150         this.load = function load() {
151             var copy = Tools.toArray(arguments);
152 
153             if (_routes.notify.apply(_routes, copy)) {
154                 copy.unshift("route");
155                 _events.notify.apply(_events, copy);
156                 return true;
157             } else {
158                 return false;
159             }
160         };
161 
162         /**
163          * Watch for route changes
164          * @param {Function} func the func to execute when the route changes
165          * @param {Object} scope the scope in which to execute the function
166          * @returns {Object} the handle to unwatch for route changes
167          */
168         this.watch = function watch(func, scope) {
169             return _events.watch("route", func, scope);
170         };
171 
172         /**
173          * Unwatch routes changes
174          * @param {Object} handle the handle was returned by the watch function
175          * @returns true if unwatch
176          */
177         this.unwatch = function unwatch(handle) {
178             return _events.unwatch(handle);
179         };
180 
181         /**
182          * Get the history store, for debugging only
183          * @private
184          */
185         this.getHistoryStore = function getHistoryStore() {
186             return _history;
187         };
188 
189         /**
190          * Get the current length of history
191          * @returns {Number} the length of history
192          */
193         this.getHistoryCount = function getHistoryCount() {
194             return _history.count();
195         };
196 
197         /**
198          * Flush the entire history
199          */
200         this.clearHistory = function clearHistory() {
201             _history.reset([]);
202         };
203 
204         /**
205          * Go back and forth in the history
206          * @param {Number} nb the amount of history to rewind/forward
207          * @returns true if history exists
208          */
209         this.go = function go(nb) {
210             var history = _history.get(_currentPos + nb);
211             if (history) {
212                 _currentPos += nb;
213                 this.load.apply(this, history);
214                 return true;
215             } else {
216                 return false;
217             }
218         };
219 
220         /**
221          * Go back in the history, short for go(-1)
222          * @returns
223          */
224         this.back = function back() {
225             return this.go(-1);
226         };
227 
228         /**
229          * Go forward in the history, short for go(1)
230          * @returns
231          */
232         this.forward = function forward() {
233             return this.go(1);
234         };
235 
236     };
237 
238 });

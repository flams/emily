  6 define(
  7 /**
  8  * @class
  9  * Transport hides and centralizes the logic behind requests.
 10  * It can issue requests to request handlers, which in turn can issue requests
 11  * to anything your node.js server has access to (HTTP, FileSystem, SIP...)
 12  */
 13 function Transport() {
 14 
 15     "use strict";
 16 
 17     /**
 18      * Create a Transport
 19      * @param {Emily Store} [optionanl] $reqHandlers an object containing the request handlers
 20      * @returns
 21      */
 22     return function TransportConstructor($reqHandlers) {
 23 
 24         /**
 25          * The request handlers
 26          * @private
 27          */
 28         var _reqHandlers = null;
 29 
 30         /**
 31          * Set the requests handlers object
 32          * @param {Emily Store} reqHandlers an object containing the requests handlers
 33          * @returns
 34          */
 35         this.setReqHandlers = function setReqHandlers(reqHandlers) {
 36             if (reqHandlers instanceof Object) {
 37                 _reqHandlers = reqHandlers;
 38                 return true;
 39             } else {
 40                 return false;
 41             }
 42         };
 43 
 44         /**
 45          * Get the requests handlers
 46          * @returns{ Emily Store} reqHandlers the object containing the requests handlers
 47          */
 48         this.getReqHandlers = function getReqHandlers() {
 49             return _reqHandlers;
 50         };
 51 
 52         /**
 53          * Issue a request to a request handler
 54          * @param {String} reqHandler the name of the request handler to issue the request to
 55          * @param {Object} data the data, or payload, to send to the request handler
 56          * @param {Function} callback the function to execute with the result
 57          * @param {Object} scope the scope in which to execute the callback
 58          * @returns
 59          */
 60         this.request = function request(reqHandler, data, callback, scope) {
 61             if (_reqHandlers.has(reqHandler) &&
 62                 typeof data != "undefined") {
 63 
 64                 _reqHandlers.get(reqHandler)(data, function () {
 65                     if (callback) {
 66                         callback.apply(scope, arguments);
 67                     }
 68                 });
 69                 return true;
 70             } else {
 71                 return false;
 72             }
 73         };
 74 
 75         /**
 76          * Issue a request to a reqHandler but keep listening for the response as it can be sent in several chunks
 77          * or remain open as long as the abort funciton is not called
 78          * @param {String} reqHandler the name of the request handler to issue the request to
 79          * @param {Object} data the data, or payload, to send to the request handler
 80          * @param {Function} callback the function to execute with the result
 81          * @param {Object} scope the scope in which to execute the callback
 82          * @returns {Function} the abort function to call to stop listening
 83          */
 84         this.listen = function listen(reqHandler, data, callback, scope) {
 85             if (_reqHandlers.has(reqHandler) &&
 86                 typeof data != "undefined" &&
 87                 typeof callback == "function") {
 88 
 89                 var func = function () {
 90                     callback.apply(scope, arguments);
 91                 },
 92                 abort;
 93 
 94                 abort = _reqHandlers.get(reqHandler)(data, func, func);
 95                 return function () {
 96                     if (typeof abort == "function") {
 97                         abort();
 98                     } else if (typeof abort == "object" && typeof abort.func == "function") {
 99                         abort.func.call(abort.scope);
100                     }
101                 };
102             } else {
103                 return false;
104             }
105         };
106 
107         this.setReqHandlers($reqHandlers);
108 
109     };
110 
111 });

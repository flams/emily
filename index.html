What is Emily? | Examples | Documentation

Emily is a JavaScript Framework.
It offers convenient tools for building MVC applications.
It's nothing else than a set of AMD modules.

Observable: the all mighty observable design pattern

	require(["Observable"], function (Observable) {
	
	 var observable = new Observable;
	 
	 observable.watch("greetings", function (greet) {
	  console.log(greet, "world!");
	 });
	 
	 observable.notify("greetings", "hello");
	
	});

Store: store data and get notified on changes

	require(["Store"], function (Store) {
	
		var store = new Store;
		
		store.set("greeting", "Hello World!");
		store.get("greeting");
	});

Promise: A simple interpretation of promises

	require(["Promise"], function (Promise) {
	
		var promise = new Promise;
		
		promise.then(function () {
			console.log("Hello World!");
		});
		
		promise.resolve();
	});

Finite State Machine: Replace your if else if then else if by a nice state machine!

	require(["StateMachine"], function (StateMachine) {
	
		var stateMachine = new StateMachine("Init", {
			"Init": [["hello", function (greeting) {
						console.log("Hello", greeting);
					}, "Greetd"]],
			"Greetd": []
		});
		
		stateMachine.event("world!");
	});
